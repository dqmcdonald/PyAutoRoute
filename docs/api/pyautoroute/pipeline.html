pyautoroute.pipeline

Shared place→route orchestration for the CLI and GUI.

Two shared units, both binding no Reporter or Tk objects (live progress flows through optional, no-op-by-default hook callbacks) so the same code backs the command line and the GUI worker, removing the orchestration that used to be duplicated between them:

  • run_pipeline — the full place(best-of-place_runs) → route(best-of-runs, optionally across --jobs worker processes) → select-best pipeline, driven by PipelineHooks. This is what autoroute.run and the GUI worker share.
  • run_cycle — one self-contained cycle (fresh load_board, one placement, one routing) scored by the routed outcome (unrouted, routed_energy); select_best keeps the lowest. Picklable, so --cycles N runs independent cycles inline or in a ProcessPoolExecutor and keeps the one that actually routes best — a signal a placement-energy proxy can't see.

The routing half of both (_route_one_run, greedy route + optional anneal) also backs the older --runs best-of-N routing loop, and is dispatched to worker processes via _route_run_worker.

 37def _route_one_run(grid, conns, order, params, run_idx, *, annealing,
 38                   iters, time_budget, seed, unrouted_weight, anneal_temps,
 39                   via_weight, on_route_progress=None, on_anneal_progress=None,
 40                   on_snapshot=None, snapshots=0, cancel=None):
 41    """Run one independent route (+ optional anneal) and return its outcome.
 42
 43    This is the shared body of the sequential and parallel best-of-N routing
 44    paths and of a `run_cycle`'s routing half. It builds a fresh `RoutingState`
 45    over `grid` (which it never mutates), greedily routes, then optionally anneals
 46    with seed ``seed + run_idx`` — matching the sequential loop's seeding exactly.
 47
 48    Args:
 49        grid: the (read-only) routing grid; each run gets its own `RoutingState`.
 50        conns: the connection list.
 51        order: the greedy routing order.
 52        params: the `RouteParams`.
 53        run_idx: zero-based run index; offsets the anneal seed.
 54        annealing: whether to run the rip-up/reroute annealer.
 55        iters: anneal iteration budget.
 56        time_budget: anneal time budget (s).
 57        seed: base seed (the anneal seed is ``seed + run_idx``).
 58        unrouted_weight: anneal unrouted-connection penalty.
 59        anneal_temps: (start, end) anneal temperatures.
 60        via_weight: via cost used for the non-annealing energy.
 61        on_route_progress: optional greedy-route progress callback.
 62        on_anneal_progress: optional anneal progress callback.
 63        on_snapshot: optional snapshot callback (single-run only).
 64        snapshots: number of snapshots to emit during annealing.
 65        cancel: optional cancellation `Event`.
 66
 67    Returns:
 68        A dict ``{energy, results, metrics, summary}`` where ``metrics`` is
 69        ``(routed, unrouted, length, vias)`` and ``summary`` is a log line (or
 70        ``None``). All values are picklable so the dict can cross a process
 71        boundary.
 72    """
 73    state = router.RoutingState(grid)
 74    result = router.route_all(state, conns, order, params,
 75                              on_progress=on_route_progress)
 76    run_results = result.results
 77    run_metrics = (result.routed, result.unrouted,
 78                   result.total_length, result.total_vias)
 79    summary = None
 80
 81    if annealing:
 82        ap = anneal.AnnealParams(iters=iters, time_budget=time_budget,
 83                                 seed=seed + run_idx, snapshots=snapshots,
 84                                 unrouted_weight=unrouted_weight,
 85                                 t_start=anneal_temps[0], t_end=anneal_temps[1],
 86                                 route_params=params)
 87        aout = anneal.anneal(state, conns, list(result.results), ap,
 88                             on_progress=on_anneal_progress,
 89                             on_snapshot=on_snapshot, cancel=cancel)
 90        run_results = aout.results
 91        run_metrics = (aout.routed, aout.unrouted,
 92                       aout.total_length, aout.total_vias)
 93        run_energy = aout.best_energy
 94        acc_pct = 100 * aout.accepted / max(aout.iterations, 1)
 95        summary = (f"anneal: {aout.iterations} iters, "
 96                   f"{aout.accepted} accepted ({acc_pct:.0f}%), "
 97                   f"energy {aout.start_energy:.1f} -> {aout.best_energy:.1f}")
 98    else:
 99        run_energy = anneal._energy(run_results, via_weight, unrouted_weight)
100
101    return {"energy": run_energy, "results": run_results,
102            "metrics": run_metrics, "summary": summary}

105def _route_run_worker(payload):
106    """Picklable `ProcessPoolExecutor` entry point for one routing run.
107
108    Unpacks a ``(grid, conns, order, params, run_idx, kwargs)`` tuple and calls
109    `_route_one_run` with all progress callbacks suppressed (per-run live
110    progress does not interleave cleanly across processes).
111
112    Args:
113        payload: the packed argument tuple.
114
115    Returns:
116        The result dict from `_route_one_run`.
117    """
118    grid, conns, order, params, run_idx, kw = payload
119    return _route_one_run(grid, conns, order, params, run_idx, **kw)

122@dataclass
123class CycleResult:
124    """Outcome of one place→route→score cycle.
125
126    Attributes:
127        seed: the cycle's seed (placement and routing both keyed off it).
128        board: the placed board, with the placement synced into its tree and the
129            outline finalised — ready to write once routing nodes are attached.
130        grid: the routing grid built over `board` (node ↔ coordinate mapping,
131            needed to flatten `results` into KiCad nodes).
132        n_conns: number of connections routed (the rats-nest size).
133        conns: the connection list (parallel to `results`); carried so congestion
134            feedback (``--place-feedback``) can find the endpoints of the
135            connections left unrouted.
136        results: per-connection `router.RouteResult` (or ``None`` for unrouted).
137        routed: connections completed.
138        unrouted: connections left open.
139        length: total routed track length (mm).
140        vias: total vias placed.
141        energy: routed energy (``length + via_weight·vias + unrouted_weight·unrouted``).
142        summary: the annealing summary line, or ``None`` when not annealed.
143
144    All fields are picklable, so a worker process can return the whole result.
145    """
146    seed: int
147    board: object
148    grid: object
149    n_conns: int
150    conns: list
151    results: list
152    routed: int
153    unrouted: int
154    length: float
155    vias: int
156    energy: float
157    summary: str | None
158
159    @property
160    def score(self) -> tuple[int, float]:
161        """Selection key: fewest unrouted first, then lowest routed energy.
162
163        Lexicographic, so completing connections always dominates — a placement
164        that routes one more net beats any that routes fewer, regardless of
165        length/vias; ties break on energy.
166        """
167        return (self.unrouted, self.energy)

170def run_cycle(input_path, rules, pitch: float, place_params, route_params, *,
171              route_kw: dict, place_margin: float, seed: int,
172              cancel=None, hooks=None) -> CycleResult:
173    """Place and route a fresh copy of a board once, and score the result.
174
175    Loads `input_path` anew (so no placement/routing state leaks between cycles),
176    places it with ``replace(place_params, seed=seed)`` — a single placement run —
177    finalises the placement into the board tree, builds the routing grid, then
178    routes (with optional annealing) via `_route_one_run` keyed off the same
179    `seed`. The returned `CycleResult` is ready to write and carries the routed
180    score.
181
182    Args:
183        input_path: the source ``.kicad_pcb`` path (re-read each call).
184        rules: the parsed design rules (for the grid).
185        pitch: routing-grid pitch in mm.
186        place_params: base `placement.PlaceParams`; the seed is overridden with
187            `seed`, the placement runs once.
188        route_params: the `router.RouteParams`.
189        route_kw: keyword dict forwarded to `_route_one_run` (``annealing``,
190            ``iters``, ``time_budget``, ``unrouted_weight``, ``anneal_temps``,
191            ``via_weight``); its own ``seed`` is supplied here.
192        place_margin: outline margin (mm) for `pcb.apply_placement`.
193        seed: this cycle's seed (placement and routing).
194        cancel: optional cancellation `Event`, threaded into place and route.
195        hooks: optional `CycleHooks` for live progress; ``None`` suppresses it
196            (the parallel-worker and headless default).
197
198    Returns:
199        The `CycleResult` for this cycle.
200    """
201    h = hooks or CycleHooks()
202    board = pcb.load_board(input_path)
203
204    pp = replace(place_params, seed=seed)
205    h.phase(f"placing {len(board.footprints)} footprints")
206    placement.place(board, pp, on_progress=h.place_progress, cancel=cancel)
207    kept = pcb.apply_placement(board, margin=place_margin,
208                               keep_outline=pp.keep_outline)
209    pcb.sync_tree_from_placement(board, keep_outline=kept)
210    h.board_snap(board)
211
212    grid = Grid(board, rules, pitch)
213    conns = netlist.build_connections(board, exclude=pp.exclude)
214    order = netlist.greedy_order(conns)
215
216    h.phase(f"routing {len(conns)} connections")
217    out = _route_one_run(grid, conns, order, route_params, 0, seed=seed,
218                         on_route_progress=h.route_progress,
219                         on_anneal_progress=h.anneal_progress,
220                         cancel=cancel, **route_kw)
221    routed, unrouted, length, vias = out["metrics"]
222    return CycleResult(seed=seed, board=board, grid=grid, n_conns=len(conns),
223                       conns=conns, results=out["results"], routed=routed,
224                       unrouted=unrouted, length=length, vias=vias,
225                       energy=out["energy"], summary=out["summary"])

247def select_best(cycles):
248    """Return the lowest-scoring `CycleResult` (see `CycleResult.score`).
249
250    Args:
251        cycles: an iterable of `CycleResult`.
252
253    Returns:
254        The cycle with the fewest unrouted connections, ties broken by routed
255        energy; ``None`` if `cycles` is empty.
256    """
257    best = None
258    for c in cycles:
259        if best is None or c.score < best.score:
260            best = c
261    return best

329    route_partial: object = None

347        energy: best routed energy (``inf`` when ``placed_only`` or nothing routed).
348        place_stats: the `placement.PlaceResult` (for reporting), or ``None``.
349        placed_only: whether routing was skipped (``--place-only``).
350        cancelled: whether the run was cancelled before any routing completed.
351    """

366def run_placement(board, *, place_params, place_runs: int, seed: int,
367                  place_margin: float, hooks=None, cancel=None):
368    """Place a board's footprints (best-of-`place_runs`) and finalise the layout.
369
370    Runs `placement.place(runs=place_runs)` with seed `seed`, then
371    `pcb.apply_placement` + `pcb.sync_tree_from_placement` so the board carries the
372    chosen placement (outline regenerated, or kept under ``keep_outline``) and is
373    ready to route or write. Mutates `board` in place. Live progress — including
374    per-run boundaries and the running overall-best — flows through `hooks`.
375
376    Args:
377        board: the parsed board, mutated in place.
378        place_params: base `placement.PlaceParams` (seed overridden with `seed`).
379        place_runs: number of placement runs (best by placement energy is kept).
380        seed: placement seed.
381        place_margin: outline margin (mm) for `pcb.apply_placement`.
382        hooks: optional `PipelineHooks`.
383        cancel: optional cancellation `Event`.
384
385    Returns:
386        The `placement.PlaceResult` (for reporting).
387    """
388    h = hooks or PipelineHooks()
389    n_fps = len(board.footprints)
390    prog = {"run": 0, "last_it": -1, "ob": float("inf")}
391
392    def on_place(it, total, energy, best, temp, accept):
393        if prog["last_it"] >= 0 and it < prog["last_it"]:
394            prog["run"] += 1
395            _call(h.place_run, prog["run"], place_runs)
396        prog["last_it"] = it
397        if best < prog["ob"]:
398            prog["ob"] = best
399        ob = prog["ob"] if place_runs > 1 else None
400        _call(h.place_progress, it, total, energy, best, temp, accept, ob)
401
402    _call(h.phase, f"placing {n_fps} footprints")
403    _call(h.place_run, 0, place_runs)
404    place_stats = placement.place(board, replace(place_params, seed=seed),
405                                  on_progress=on_place, runs=place_runs,
406                                  cancel=cancel)
407    kept = pcb.apply_placement(board, margin=place_margin,
408                               keep_outline=place_params.keep_outline)
409    pcb.sync_tree_from_placement(board, keep_outline=kept)
410    _call(h.placed, board)
411    return place_stats

414def run_routing(board, rules, pitch: float, *, route_params, route_kw: dict,
415                seed: int, runs: int, jobs: int, snapshots: int, exclude,
416                grid=None, conns=None, order=None,
417                hooks=None, cancel=None) -> PipelineResult:
418    """Route a (placed) board best-of-`runs` and keep the lowest-energy result.
419
420    Builds the connections, greedy order, and grid over `board` (unless passed in
421    pre-built — the CLI builds them early for its parameter dump), then routes
422    `runs` times (each greedy route + optional anneal). With ``runs > 1`` and
423    ``jobs > 1`` the runs are dispatched across a `ProcessPoolExecutor` (progress
424    suppressed); otherwise they run sequentially with live `hooks`. Selection —
425    lowest routed energy — matches both paths.
426
427    Args:
428        board: the placed board to route.
429        rules: the design rules (for the grid).
430        pitch: routing-grid pitch (mm).
431        route_params: the `router.RouteParams`.
432        route_kw: ``annealing``/``iters``/``time_budget``/``unrouted_weight``/
433            ``anneal_temps``/``via_weight`` (the seed is the separate `seed` arg).
434        seed: base seed (routing run ``k`` keys off ``seed + k``).
435        runs: number of routing runs (best kept).
436        jobs: worker processes for parallel routing (``> 1`` enables it).
437        snapshots: annealing snapshot count (single-run only; 0 disables).
438        exclude: net patterns to leave unrouted.
439        grid: optional pre-built routing grid (built here, with a phase event, if
440            ``None``).
441        conns: optional pre-built connection list (built here if ``None``).
442        order: optional pre-built greedy order (built here if ``None``).
443        hooks: optional `PipelineHooks`.
444        cancel: optional cancellation `Event`.
445
446    Returns:
447        The `PipelineResult` (``place_stats`` is ``None``; the caller fills it).
448    """
449    h = hooks or PipelineHooks()
450    if conns is None:
451        _call(h.phase, "building netlist (MST rats-nest)")
452        conns = netlist.build_connections(board, exclude=exclude or [])
453    if order is None:
454        order = netlist.greedy_order(conns)
455    if grid is None:
456        _call(h.phase, f"building {pitch}mm routing grid")
457        grid = Grid(board, rules, pitch)
458    annealing = route_kw["annealing"]
459
460    best_energy = float("inf")
461    final_results = None
462    routed = unrouted = length = vias = 0
463    parallel = runs > 1 and jobs > 1
464
465    if parallel:
466        _call(h.phase, f"routing {len(conns)} connections — "
467                       f"{runs} runs across {jobs} workers")
468        kw = dict(route_kw, seed=seed)
469        payloads = [(grid, conns, order, route_params, k, kw) for k in range(runs)]
470        done_n = 0
471        try:
472            with concurrent.futures.ProcessPoolExecutor(max_workers=jobs) as ex:
473                futs = [ex.submit(_route_run_worker, p) for p in payloads]
474                for fut in concurrent.futures.as_completed(futs):
475                    out = fut.result()
476                    _call(h.route_run_done, done_n, runs, out["energy"],
477                          out["summary"], out["metrics"])
478                    done_n += 1
479                    if out["energy"] < best_energy:
480                        best_energy = out["energy"]
481                        final_results = out["results"]
482                        routed, unrouted, length, vias = out["metrics"]
483        except KeyboardInterrupt:
484            if final_results is None:
485                raise
486    else:
487        for k in range(runs):
488            if cancel is not None and cancel.is_set():
489                break
490            _call(h.route_run, k, runs)
491            _call(h.phase, f"routing {len(conns)} connections")
492            partial = [None] * len(conns)
493
494            def on_partial(idx, res):
495                partial[idx] = res
496                _call(h.route_partial, board, grid, partial)
497
498            state = router.RoutingState(grid)
499            result = router.route_all(
500                state, conns, order, route_params,
501                on_progress=h.route_progress,
502                on_partial=on_partial if h.route_partial is not None else None)
503            run_results = result.results
504            run_metrics = (result.routed, result.unrouted,
505                           result.total_length, result.total_vias)
506            summary = None
507
508            if annealing:
509                _call(h.phase, "annealing (rip-up & reroute)")
510                ob_e = best_energy if runs > 1 and best_energy < float("inf") else None
511
512                def on_anneal(it, total, r, u, energy, best, temp, accept):

524                    seed=seed + k, snapshots=snapshots,
525                    unrouted_weight=route_kw["unrouted_weight"],
526                    t_start=route_kw["anneal_temps"][0],
527                    t_end=route_kw["anneal_temps"][1], route_params=route_params)
528                aout = anneal.anneal(
529                    state, conns, list(result.results), ap,
530                    on_progress=on_anneal if h.anneal_progress is not None else None,
531                    on_snapshot=on_snap if snapshots else None,
532                    cancel=cancel,
533                    on_best=on_best if h.anneal_best is not None else None)
534                run_results = aout.results
535                run_metrics = (aout.routed, aout.unrouted,

416                seed: int, runs: int, jobs: int, snapshots: int, exclude,
417                grid=None, conns=None, order=None,
418                hooks=None, cancel=None) -> PipelineResult:
419    """Route a (placed) board best-of-`runs` and keep the lowest-energy result.
420
421    Builds the connections, greedy order, and grid over `board` (unless passed in
422    pre-built — the CLI builds them early for its parameter dump), then routes
423    `runs` times (each greedy route + optional anneal). With ``runs > 1`` and
424    ``jobs > 1`` the runs are dispatched across a `ProcessPoolExecutor` (progress
425    suppressed); otherwise they run sequentially with live `hooks`. Selection —
426    lowest routed energy — matches both paths.
427
428    Args:
429        board: the placed board to route.
430        rules: the design rules (for the grid).
431        pitch: routing-grid pitch (mm).
432        route_params: the `router.RouteParams`.
433        route_kw: ``annealing``/``iters``/``time_budget``/``unrouted_weight``/
434            ``anneal_temps``/``via_weight`` (the seed is the separate `seed` arg).
435        seed: base seed (routing run ``k`` keys off ``seed + k``).
436        runs: number of routing runs (best kept).
437        jobs: worker processes for parallel routing (``> 1`` enables it).
438        snapshots: annealing snapshot count (single-run only; 0 disables).
439        exclude: net patterns to leave unrouted.
440        grid: optional pre-built routing grid (built here, with a phase event, if
441            ``None``).
442        conns: optional pre-built connection list (built here if ``None``).
443        order: optional pre-built greedy order (built here if ``None``).
444        hooks: optional `PipelineHooks`.
445        cancel: optional cancellation `Event`.
446
447    Returns:
448        The `PipelineResult` (``place_stats`` is ``None``; the caller fills it).
449    """
450    h = hooks or PipelineHooks()
451    if conns is None:
452        _call(h.phase, "building netlist (MST rats-nest)")
453        conns = netlist.build_connections(board, exclude=exclude or [])
454    if order is None:
455        order = netlist.greedy_order(conns)
456    if grid is None:
457        _call(h.phase, f"building {pitch}mm routing grid")
458        grid = Grid(board, rules, pitch)
459    annealing = route_kw["annealing"]
460
461    best_energy = float("inf")
462    final_results = None
463    routed = unrouted = length = vias = 0
464    parallel = runs > 1 and jobs > 1
465
466    if parallel:
467        _call(h.phase, f"routing {len(conns)} connections — "
468                       f"{runs} runs across {jobs} workers")
469        kw = dict(route_kw, seed=seed)
470        payloads = [(grid, conns, order, route_params, k, kw) for k in range(runs)]
471        done_n = 0
472        try:
473            with concurrent.futures.ProcessPoolExecutor(max_workers=jobs) as ex:
474                futs = [ex.submit(_route_run_worker, p) for p in payloads]
475                for fut in concurrent.futures.as_completed(futs):
476                    out = fut.result()
477                    _call(h.route_run_done, done_n, runs, out["energy"],
478                          out["summary"], out["metrics"])
479                    done_n += 1
480                    if out["energy"] < best_energy:
481                        best_energy = out["energy"]
482                        final_results = out["results"]
483                        routed, unrouted, length, vias = out["metrics"]
484        except KeyboardInterrupt:
485            if final_results is None:
486                raise
487    else:
488        for k in range(runs):
489            if cancel is not None and cancel.is_set():
490                break
491            _call(h.route_run, k, runs)
492            _call(h.phase, f"routing {len(conns)} connections")
493            partial = [None] * len(conns)
494
495            def on_partial(idx, res):
496                partial[idx] = res
497                _call(h.route_partial, board, grid, partial)
498
499            state = router.RoutingState(grid)
500            result = router.route_all(
501                state, conns, order, route_params,
502                on_progress=h.route_progress,
503                on_partial=on_partial if h.route_partial is not None else None)
504            run_results = result.results
505            run_metrics = (result.routed, result.unrouted,
506                           result.total_length, result.total_vias)
507            summary = None
508
509            if annealing:
510                _call(h.phase, "annealing (rip-up & reroute)")
511                ob_e = best_energy if runs > 1 and best_energy < float("inf") else None
512
513                def on_anneal(it, total, r, u, energy, best, temp, accept):
514                    _call(h.anneal_progress, it, total, r, u, energy, best, temp,
515                          accept, ob_e)
516
517                def on_snap(kk, nn, res):
518                    _call(h.anneal_snapshot, board, grid, res, kk, nn)
519
520                def on_best(be, br):
521                    _call(h.anneal_best, board, grid, br)
522
523                ap = anneal.AnnealParams(
524                    iters=route_kw["iters"], time_budget=route_kw["time_budget"],
525                    seed=seed + k, snapshots=snapshots,
526                    unrouted_weight=route_kw["unrouted_weight"],
527                    t_start=route_kw["anneal_temps"][0],
528                    t_end=route_kw["anneal_temps"][1], route_params=route_params)
529                aout = anneal.anneal(
530                    state, conns, list(result.results), ap,
531                    on_progress=on_anneal if h.anneal_progress is not None else None,
532                    on_snapshot=on_snap if snapshots else None,
533                    cancel=cancel,
534                    on_best=on_best if h.anneal_best is not None else None)
535                run_results = aout.results
536                run_metrics = (aout.routed, aout.unrouted,
537                               aout.total_length, aout.total_vias)
538                run_energy = aout.best_energy
539                acc_pct = 100 * aout.accepted / max(aout.iterations, 1)
540                summary = (f"anneal: {aout.iterations} iters, "
541                           f"{aout.accepted} accepted ({acc_pct:.0f}%), "
542                           f"energy {aout.start_energy:.1f} -> {aout.best_energy:.1f}")
543            else:
544                run_energy = anneal._energy(run_results, route_kw["via_weight"],
545                                            route_kw["unrouted_weight"])
546
547            _call(h.route_run_done, k, runs, run_energy, summary, run_metrics)
548            if run_energy < best_energy:
549                best_energy = run_energy
550                final_results = run_results
551                routed, unrouted, length, vias = run_metrics
552                _call(h.overall_best, board, grid, final_results, best_energy)
553
554    return PipelineResult(board=board, grid=grid, n_conns=len(conns),
555                          results=final_results, routed=routed, unrouted=unrouted,
556                          length=length, vias=vias, energy=best_energy,
557                          place_stats=None, placed_only=False,
558                          cancelled=(final_results is None))
