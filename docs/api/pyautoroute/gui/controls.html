 41class _ScrollFrame(ttk.Frame):
 42    """A vertically scrollable frame — inner content goes in ``self.inner``."""
 43
 44    def __init__(self, parent, **kw):
 45        super().__init__(parent, **kw)
 46        vbar = ttk.Scrollbar(self, orient=tk.VERTICAL)
 47        self._canvas = tk.Canvas(self, yscrollcommand=vbar.set,
 48                                  highlightthickness=0)
 49        vbar.config(command=self._canvas.yview)
 50        vbar.pack(side=tk.RIGHT, fill=tk.Y)
 51        self._canvas.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)
 52        self.inner = ttk.Frame(self._canvas)
 53        self._win_id = self._canvas.create_window(
 54            (0, 0), window=self.inner, anchor=tk.NW)
 55        self.inner.bind("<Configure>", self._on_inner)
 56        self._canvas.bind("<Configure>", self._on_canvas)
 57        self._canvas.bind("<MouseWheel>", self._on_scroll)
 58        self._canvas.bind("<Button-4>", self._on_scroll)
 59        self._canvas.bind("<Button-5>", self._on_scroll)
 60
 61    def _on_inner(self, _e=None):
 62        self._canvas.configure(
 63            scrollregion=self._canvas.bbox(tk.ALL))
 64
 65    def _on_canvas(self, e):
 66        self._canvas.itemconfig(self._win_id, width=e.width)
 67
 68    def _on_scroll(self, e):
 69        if e.num == 4:
 70            self._canvas.yview_scroll(-1, "units")
 71        elif e.num == 5:
 72            self._canvas.yview_scroll(1, "units")
 73        else:
 74            self._canvas.yview_scroll(int(-1 * e.delta / 120), "units")

347    def set_input(self, path: str) -> None:
348        p = Path(path)
349        self._input_path.set(p.name)
350        # Auto-detect .pro
351        pro = p.with_suffix(".kicad_pro")
352        if not pro.exists():
353            pro = p.with_name(p.stem + ".kicad_pro")
354        self._pro_path.set(pro.name if pro.exists() else "(not found)")
355        self._full_input = str(p)
356        # Auto-load project config: prefer <stem>.ini, fall back to .pyautoroute.cfg
357        for cfg in (p.with_suffix(".ini"),
358                    p.with_name(p.stem + ".pyautoroute.cfg")):
359            if cfg.exists():
360                self._load_cfg(str(cfg))
361                break

368    def _save_settings(self):
369        inp = self._full_input_path()
370        init = str(Path(inp).with_name(
371            Path(inp).stem + ".pyautoroute.cfg")) if inp else ""
372        path = filedialog.asksaveasfilename(
373            title="Save settings",
374            initialfile=Path(init).name if init else "",
375            initialdir=str(Path(init).parent) if init else ".",
376            defaultextension=".cfg",
377            filetypes=[("Config", "*.cfg *.pyautoroute.cfg"),
378                       ("All files", "*.*")])
379        if not path:
380            return
381        try:
382            from pyautoroute.autoroute import build_parser, write_config
383            parser = build_parser()
384            args = self._to_namespace(parser)
385            write_config(parser, args, path)
386        except Exception as exc:
387            messagebox.showerror("Save failed", str(exc))

397    def _load_cfg(self, path: str) -> None:
398        try:
399            from pyautoroute.autoroute import build_parser, load_config
400            parser = build_parser()
401            # provide a dummy input so the parser doesn't complain
402            d = load_config(path, parser)
403            self._apply_dict(d)
404        except Exception as exc:
405            messagebox.showerror("Load failed", str(exc))

407    def _apply_dict(self, d: dict) -> None:
408        """Apply a {dest: value} dict (from load_config) to the UI vars."""
409        if "place" in d and d["place"]:
410            self._mode.set("place_route")
411        if "place_only" in d and d["place_only"]:
412            self._mode.set("place_only")
413        _sv = lambda key, var: var.set(str(d[key])) if key in d and d[key] is not None else None
414        _sv("grid", self._grid)
415        _sv("runs", self._runs)
416        if "iters" in d and d["iters"]:
417            self._budget_kind.set("iters")
418            self._budget_val.set(str(d["iters"]))
419        if "time_budget" in d and d["time_budget"]:
420            self._budget_kind.set("time")
421            self._budget_val.set(str(d["time_budget"]))
422        if "exclude_net" in d and d["exclude_net"]:
423            self._exclude_net.set(", ".join(d["exclude_net"]))
424        _sv("place_runs", self._place_runs)
425        _sv("place_margin", self._place_margin)
426        _sv("place_buffer", self._place_buffer)
427        if "place_rotate" in d and d["place_rotate"]:
428            self._place_rotate.set(d["place_rotate"])
429        if "place_iters" in d and d["place_iters"]:
430            self._place_budget_kind.set("iters")
431            self._place_budget_val.set(str(d["place_iters"]))
432        if "place_time" in d and d["place_time"]:
433            self._place_budget_kind.set("time")
434            self._place_budget_val.set(str(d["place_time"]))
435        _sv("seed", self._seed)
436        _sv("via_weight", self._via_weight)
437        _sv("unrouted_weight", self._unrouted_weight)
438        if "anneal_temps" in d and d["anneal_temps"]:
439            self._anneal_t_start.set(str(d["anneal_temps"][0]))
440            self._anneal_t_end.set(str(d["anneal_temps"][1]))
441        if "place_temps" in d and d["place_temps"]:
442            self._place_t_start.set(str(d["place_temps"][0]))
443            self._place_t_end.set(str(d["place_temps"][1]))
444        _sv("place_step", self._place_step)
445        _sv("place_overlap_weight", self._place_ow)
446        _sv("place_compact_weight", self._place_cw)
447        _sv("auto_probe_time", self._auto_probe_time)
448        if "debug_plot" in d:
449            self._debug_plot.set(bool(d["debug_plot"]))
450        if "quiet" in d:
451            self._quiet.set(bool(d["quiet"]))
452        if "fix_values" in d:
453            self._fix_values.set(bool(d["fix_values"]))
