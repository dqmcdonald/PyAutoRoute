 32    def __init__(self, parent, **kw):
 33        super().__init__(parent, **kw)
 34        self._vars: dict[str, tk.StringVar] = {}
 35        rows = [
 36            ("phase",    "Phase:",      "Current pipeline phase"),
 37            ("elapsed",  "Elapsed:",    "Wall-clock time since Run was pressed"),
 38            ("iter",     "Iteration:",  "Current / total iterations"),
 39            ("energy",   "Energy:",     "Current / best energy"),
 40            ("accept",   "Accept %:",   "Fraction of recent annealing moves accepted"),
 41            ("temp",     "Temp:",       "Current annealing temperature"),
 42            ("routed",   "Routed:",     "Connections routed so far"),
 43            ("unrouted", "Unrouted:",   "Connections that could not be routed"),
 44            ("check",    "Self-check:", "DRC clearance violations after routing"),
 45        ]
 46        for i, (key, lbl, tip) in enumerate(rows):
 47            var = tk.StringVar(value="—")
 48            self._vars[key] = var
 49            ttk.Label(self, text=lbl, anchor=tk.E).grid(
 50                row=i, column=0, sticky=tk.E, padx=(6, 2), pady=2)
 51            ttk.Label(self, textvariable=var, anchor=tk.W).grid(
 52                row=i, column=1, sticky=tk.W, padx=(2, 6), pady=2)
 53        self.columnconfigure(1, weight=1)
 54
 55        self._pb_var = tk.IntVar(value=0)
 56        pb = ttk.Progressbar(self, variable=self._pb_var, maximum=100,
 57                             orient=tk.HORIZONTAL, length=180, mode="determinate")
 58        pb.grid(row=len(rows), column=0, columnspan=2,
 59                padx=6, pady=(4, 2), sticky=tk.EW)
 60        self._pb = pb
 61
 62        self._t0: float | None = None

 64    def reset(self) -> None:
 65        for var in self._vars.values():
 66            var.set("—")
 67        self._pb_var.set(0)
 68        self._t0 = time.monotonic()

 70    def set_phase(self, name: str) -> None:
 71        self._vars["phase"].set(name)
 72
 73    def update(self, ev: Progress) -> None:
 74        if self._t0 is not None:
 75            self._vars["elapsed"].set(f"{time.monotonic() - self._t0:.1f}s")
 76        if ev.budget > 0:
 77            pct = min(100, int(100 * ev.elapsed / ev.budget))
 78            remaining = max(0.0, ev.budget - ev.elapsed)
 79            self._vars["iter"].set(f"{remaining:.0f}s remaining")
 80        else:
 81            total = ev.total or 1
 82            pct = min(100, int(100 * ev.it / total))
 83            self._vars["iter"].set(f"{ev.it} / {ev.total}")
 84        self._pb_var.set(pct)
 85        if ev.kind in ("annealing", "placing"):
 86            self._vars["energy"].set(f"{ev.energy:.1f} / {ev.best:.1f}")
 87            self._vars["accept"].set(f"{ev.accept * 100:.0f}%")
 88            self._vars["temp"].set(f"{ev.temp:.3f}")
 89        if ev.kind in ("routing", "annealing"):
 90            self._vars["routed"].set(str(ev.routed))
 91            self._vars["unrouted"].set(str(ev.unrouted))
 92
 93    def set_done(self, ev: Done) -> None:
 94        if self._t0 is not None:
 95            self._vars["elapsed"].set(f"{time.monotonic() - self._t0:.1f}s")
 96        self._pb_var.set(100)
 97        self._vars["routed"].set(str(ev.routed))
 98        self._vars["unrouted"].set(str(ev.unrouted))
 99        n_viol = len(ev.violations)
100        self._vars["check"].set("PASS" if n_viol == 0
101                                else f"{n_viol} violation(s)")
102
103    def set_initial_stats(self, stats) -> None:
104        """Populate the panel with pre-run board routing statistics."""
105        for key in ("elapsed", "iter", "energy", "accept", "temp"):
106            self._vars[key].set("—")
107        self._pb_var.set(0)
108        self._t0 = None
109        self._vars["phase"].set("Initial board state")
110        self._vars["routed"].set(
111            f"{stats.routed}/{stats.total}  ({stats.length:.1f} mm, {stats.vias} vias)")
112        self._vars["unrouted"].set(str(stats.unrouted))
113        n_viol = len(stats.violations)
114        self._vars["check"].set("PASS" if n_viol == 0
115                                else f"{n_viol} violation(s)")

225    def _drain(self) -> None:
226        # Drain all pending events, but collapse redundant ones so the main
227        # thread is never overwhelmed even if the worker posts rapidly:
228        #   - keep only the last BoardSnap (rendering is expensive)
229        #   - keep only the last Progress per kind (metrics just need latest)
230        #   - keep all Phase / Done / Error events (rare, always process)
231        # Wrapped in try/finally so a bug here never silently kills the loop.
232        try:
233            events: list = []
234            try:
235                while True:
236                    events.append(self._queue.get_nowait())
237            except queue.Empty:
238                pass
239
240            # Identify last BoardSnap and last Progress per kind.
241            last_snap_idx: int | None = None
242            last_prog_idx: dict[str, int] = {}
243            for i, ev in enumerate(events):
244                if isinstance(ev, BoardSnap):
245                    last_snap_idx = i
246                elif isinstance(ev, Progress):
247                    last_prog_idx[ev.kind] = i
248
249            for i, ev in enumerate(events):
250                if isinstance(ev, BoardSnap) and i != last_snap_idx:
251                    continue
252                if isinstance(ev, Progress) and last_prog_idx.get(ev.kind) != i:
253                    continue
254                try:
255                    self._handle_event(ev)
256                except Exception:
257                    pass  # don't let one bad event kill the drain loop
258
259            # Update elapsed time label while running
260            if self._worker is not None and not self._worker.join(0):
261                elapsed = time.monotonic() - self._t_run_start
262                self._metrics._vars["elapsed"].set(f"{elapsed:.1f}s")
263
264            # Flush pending widget redraws (StringVar updates, geometry).
265            # Without this, Tkinter timer events starve idle-queue repaints
266            # and labels appear frozen even though their StringVars are set.
267            self._root.update_idletasks()
268        except Exception:
269            pass  # keep the drain loop alive no matter what
270        finally:
271            self._root.after(_DRAIN_MS, self._drain)
272
273    def _handle_event(self, event) -> None:
274        if isinstance(event, Phase):
275            self._status_var.set(event.name)
276            self._metrics.set_phase(event.name)
277        elif isinstance(event, Progress):
278            self._metrics.update(event)
279            if event.kind in ("placing", "annealing"):
280                self._energy_plot.add_point(event.it, event.energy, event.best)
281                now = time.monotonic()
282                if now - self._last_plot_refresh > _PLOT_REFRESH_S:
283                    self._energy_plot.refresh()
284                    self._last_plot_refresh = now
285        elif isinstance(event, BoardSnap):
286            self._board_canvas.show_board(
287                event.board, event.results, event.grid)
288        elif isinstance(event, Done):
289            self._on_done(event)
290        elif isinstance(event, Error):
291            self._on_error(event)

295    def _start_run(self, cfg) -> None:
296        if self._worker and not self._worker.join(0):
297            return  # already running
298        self._cancel.clear()
299        self._worker = Worker(self._queue, self._cancel)
300        self._controls.set_running(True)
301        self._controls.set_apply_enabled(False)
302        self._last_done = None
303        self._show_initial.set(False)
304        self._toggle_cb.configure(state=tk.DISABLED)
305        self._toggle_label.configure(text="")
306        self._metrics.reset()
307        self._energy_plot.reset()
308        self._t_run_start = time.monotonic()
309        self._status_var.set("Starting…")
310        self._worker.start(cfg)
311
312    def _cancel_run(self) -> None:
313        self._cancel.set()
314        self._status_var.set("Stopping (keeping best so far)…")
315
316    def _on_done(self, ev: Done) -> None:
317        self._last_done = ev
318        self._controls.set_running(False)
319        self._controls.set_apply_enabled(True)
320        self._energy_plot.refresh()
321        self._metrics.set_done(ev)
322        n_viol = len(ev.violations)
323        check = "PASS" if n_viol == 0 else f"{n_viol} clearance violation(s)"
324        msg = (f"Done — {ev.routed}/{ev.total} routed, "
325               f"{ev.length:.0f} mm, {ev.vias} vias.  {check}")
326        self._status_var.set(msg)
327        # Enable initial/final toggle when we have both states
328        if self._initial_board is not None:
329            self._toggle_cb.configure(state=tk.NORMAL)
330            if self._initial_stats is not None:
331                s = self._initial_stats
332                self._toggle_label.configure(
333                    text=f"Initial: {s.routed}/{s.total} routed, "
334                         f"{s.length:.0f} mm, {s.vias} vias")
335        if n_viol:
336            messagebox.showwarning("Self-check",
337                                   f"{n_viol} clearance violation(s) found.\n"
338                                   f"Review the board in KiCad.")
339
340    def _on_error(self, ev: Error) -> None:
341        self._controls.set_running(False)
342        self._status_var.set(f"Error: {ev.exc}")
343        messagebox.showerror("Pipeline error",

348    def _apply_to_project(self) -> None:
349        if self._last_done is None:
350            return
351        out_path = Path(self._last_done.out_path)
352        inp = self._controls._full_input_path()
353        if not inp:
354            return
355        orig = Path(inp)
356        if not orig.exists():
357            messagebox.showerror("Apply failed",
358                                 f"Original not found: {orig}")
359            return
360
361        # Confirm
362        if not messagebox.askyesno(
363                "Apply to project",
364                f"Replace\n  {orig.name}\nwith the routed result?\n\n"
365                f"A backup will be created."):
366            return
367
368        # Create timestamped backup
369        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
370        bak = orig.with_suffix(f".{ts}.kicad_pcb.bak")
371        try:
372            shutil.copy2(orig, bak)
373            shutil.copy2(out_path, orig)
374        except OSError as exc:
375            messagebox.showerror("Apply failed", str(exc))
376            return
377
378        messagebox.showinfo("Applied",
379                            f"Replaced {orig.name}.\n"
380                            f"Backup: {bak.name}")
381        self._status_var.set(f"Applied — backup: {bak.name}")

400    def _open_board(self, path: str) -> None:
401        try:
402            from pyautoroute import pcb
403            from pyautoroute.report import routing_stats
404            from pyautoroute.rules import load_rules
405            board = pcb.load_board(path)
406            self._initial_board = board
407            self._initial_stats = None
408            self._last_done = None
409            self._show_initial.set(False)
410            self._toggle_cb.configure(state=tk.DISABLED)
411            self._toggle_label.configure(text="")
412
413            self._board_canvas.show_board(board, title=Path(path).name)
414
415            outline_note = (
416                "  ⚠ No Edge.Cuts found — default outline added."
417                if board.outline_synthesized else ""
418            )
419            base_msg = (
420                f"Opened {Path(path).name} — "
421                f"{len(board.pads)} pads, {len(board.copper_layers)} Cu layers"
422                f"{outline_note}"
423            )
424
425            if board.segments:
426                pro_path = Path(path).with_suffix(".kicad_pro")
427                if not pro_path.exists():
428                    pro_path = Path(path).with_name(Path(path).stem + ".kicad_pro")
429                try:
430                    rules = load_rules(pro_path)
431                except Exception:
432                    rules = None
433                stats = routing_stats(board, rules)
434                self._initial_stats = stats
435                self._metrics.set_initial_stats(stats)
436                self._status_var.set(f"{base_msg}  —  initial: {stats.summary()}")
437            else:
438                self._status_var.set(base_msg)
439        except Exception as exc:
440            messagebox.showerror("Open failed", str(exc))

442    def _on_view_toggle(self) -> None:
443        """Switch the canvas between initial and final board state."""
444        inp = self._controls._full_input_path()
445        title_base = Path(inp).name if inp else "board"
446        if self._show_initial.get():
447            if self._initial_board is not None:
448                self._board_canvas.show_board(
449                    self._initial_board, title=f"{title_base} (initial)")
450            if self._initial_stats is not None:
451                self._metrics.set_initial_stats(self._initial_stats)
452        else:
453            ev = self._last_done
454            if ev is not None and ev.board is not None:
455                self._board_canvas.show_board(ev.board, title=f"{title_base} (routed)")
456                self._metrics.set_done(ev)

224    # ── queue drain ───────────────────────────────────────────────────
225
226    def _drain(self) -> None:
227        # Drain all pending events, but collapse redundant ones so the main
228        # thread is never overwhelmed even if the worker posts rapidly:
229        #   - keep only the last BoardSnap (rendering is expensive)
230        #   - keep only the last Progress per kind (metrics just need latest)
231        #   - keep all Phase / Done / Error events (rare, always process)
232        # Wrapped in try/finally so a bug here never silently kills the loop.
233        try:
234            events: list = []
235            try:
236                while True:
237                    events.append(self._queue.get_nowait())
238            except queue.Empty:
239                pass
240
241            # Identify last BoardSnap and last Progress per kind.
242            last_snap_idx: int | None = None
243            last_prog_idx: dict[str, int] = {}
244            for i, ev in enumerate(events):
245                if isinstance(ev, BoardSnap):
246                    last_snap_idx = i
247                elif isinstance(ev, Progress):
248                    last_prog_idx[ev.kind] = i
249
250            for i, ev in enumerate(events):
251                if isinstance(ev, BoardSnap) and i != last_snap_idx:
252                    continue
253                if isinstance(ev, Progress) and last_prog_idx.get(ev.kind) != i:
254                    continue
255                try:
256                    self._handle_event(ev)
257                except Exception:
258                    pass  # don't let one bad event kill the drain loop
259
260            # Update elapsed time label while running
261            if self._worker is not None and not self._worker.join(0):
262                elapsed = time.monotonic() - self._t_run_start
263                self._metrics._vars["elapsed"].set(f"{elapsed:.1f}s")
264
265            # Flush pending widget redraws (StringVar updates, geometry).
266            # Without this, Tkinter timer events starve idle-queue repaints
267            # and labels appear frozen even though their StringVars are set.
268            self._root.update_idletasks()
269        except Exception:
270            pass  # keep the drain loop alive no matter what
271        finally:
272            self._root.after(_DRAIN_MS, self._drain)

274    def _handle_event(self, event) -> None:
275        if isinstance(event, Phase):
276            self._status_var.set(event.name)
277            self._metrics.set_phase(event.name)
278        elif isinstance(event, Progress):
279            self._metrics.update(event)
280            if event.kind in ("placing", "annealing"):
281                self._energy_plot.add_point(event.it, event.energy, event.best)
282                now = time.monotonic()
283                if now - self._last_plot_refresh > _PLOT_REFRESH_S:
284                    self._energy_plot.refresh()
285                    self._last_plot_refresh = now
286        elif isinstance(event, BoardSnap):
287            self._board_canvas.show_board(
288                event.board, event.results, event.grid)
289        elif isinstance(event, Done):
290            self._on_done(event)
291        elif isinstance(event, Error):
292            self._on_error(event)

307        self._metrics.reset()

317    def _on_done(self, ev: Done) -> None:
318        self._last_done = ev
319        self._controls.set_running(False)
320        self._controls.set_apply_enabled(True)
321        self._energy_plot.refresh()
322        self._metrics.set_done(ev)
323        n_viol = len(ev.violations)
324        check = "PASS" if n_viol == 0 else f"{n_viol} clearance violation(s)"
325        msg = (f"Done — {ev.routed}/{ev.total} routed, "
326               f"{ev.length:.0f} mm, {ev.vias} vias.  {check}")
327        self._status_var.set(msg)
328        # Enable initial/final toggle when we have both states
329        if self._initial_board is not None:
330            self._toggle_cb.configure(state=tk.NORMAL)
331            if self._initial_stats is not None:
332                s = self._initial_stats
333                self._toggle_label.configure(
334                    text=f"Initial: {s.routed}/{s.total} routed, "
335                         f"{s.length:.0f} mm, {s.vias} vias")
336        if n_viol:
337            messagebox.showwarning("Self-check",
338                                   f"{n_viol} clearance violation(s) found.\n"
339                                   f"Review the board in KiCad.")

349    def _apply_to_project(self) -> None:
350        if self._last_done is None:
351            return
352        out_path = Path(self._last_done.out_path)
353        inp = self._controls._full_input_path()
354        if not inp:
355            return
356        orig = Path(inp)
357        if not orig.exists():
358            messagebox.showerror("Apply failed",
359                                 f"Original not found: {orig}")
360            return
361
362        # Confirm
363        if not messagebox.askyesno(
364                "Apply to project",
365                f"Replace\n  {orig.name}\nwith the routed result?\n\n"
366                f"A backup will be created."):
367            return
368
369        # Create timestamped backup
370        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
371        bak = orig.with_suffix(f".{ts}.kicad_pcb.bak")
372        try:
373            shutil.copy2(orig, bak)
374            shutil.copy2(out_path, orig)
375        except OSError as exc:
376            messagebox.showerror("Apply failed", str(exc))
377            return
378
379        messagebox.showinfo("Applied",
380                            f"Replaced {orig.name}.\n"
381                            f"Backup: {bak.name}")
382        self._status_var.set(f"Applied — backup: {bak.name}")
