pyautoroute.gui.worker

Background worker thread: runs the place/route pipeline and posts events.

The worker's only interaction with the UI is to push immutable event objects onto a queue.Queue. All board mutations happen here; the snapshot copies placed in BoardSnap events are made while the board is in a consistent state so the main thread can read them safely.

The actual place→route orchestration is the shared pyautoroute.pipeline (run_placement / run_routing) — the same code the CLI runs — driven here through a pipeline.PipelineHooks whose callbacks translate live progress into the GUI's Phase / Progress / BoardSnap events (throttled so the UI queue isn't flooded). This worker only owns parsing, the GUI event translation, and writing the result.

 29# Minimum seconds between Progress events posted to the UI queue.
 30# Keeps the main thread from being overwhelmed by high-iteration-rate callbacks.
 31_PROGRESS_INTERVAL = 0.125   # ~8 Hz max
 32# Minimum seconds between board canvas snapshots during greedy routing.
 33_ROUTE_SNAP_INTERVAL = 1.5
 34# Routing annealing snapshot count for live board updates.
 35_ANNEAL_SNAP_COUNT = 40

 38def _snap_pads(board, margin: float = 2.0):
 39    """Return a Board copy frozen at a consistent instant for live placement.
 40
 41    Both ``board.pads`` and ``board.footprints`` are copied so the snapshot is an
 42    internally consistent moment: ``draw_board`` positions footprint outlines from
 43    the footprint pose (``fp.x/y/angle``) while pads come from ``board.pads``.  If
 44    only the pads were frozen, the still-running placer would keep moving the
 45    shared footprint poses, so the outlines would render offset from their pads.
 46
 47    The snapshot's outline is also regenerated to bound the current pads (the same
 48    rectangle ``apply_placement`` produces), so the autoscaled view tracks the
 49    footprints as they compress instead of staying framed to the original, larger
 50    board edge.
 51
 52    Args:
 53        board: the live board being placed.
 54        margin: extra space (mm) around the pads when sizing the preview outline;
 55            pass the placement margin so the preview matches the final outline.
 56    """
 57    from pyautoroute import pcb
 58
 59    pads_copy = [dataclasses.replace(p) for p in board.pads]
 60    fps_copy = [dataclasses.replace(fp) for fp in board.footprints]
 61    snap = dataclasses.replace(board, pads=pads_copy, footprints=fps_copy)
 62    if pads_copy:
 63        snap.outline = pcb.pad_bounding_outline(pads_copy, margin)
 64    return snap

 67class Worker:
 68    """Runs the pipeline in a daemon thread; cancel via ``cancel_event``."""

 70    def __init__(self, event_queue: queue.Queue,
 71                 cancel_event: threading.Event):
 72        self._q = event_queue
 73        self._cancel = cancel_event
 74        self._thread: threading.Thread | None = None
 75
 76    def start(self, cfg) -> None:
 77        """Launch the pipeline for the given ``RunConfig`` *cfg*."""
 78        self._thread = threading.Thread(
 79            target=self._run, args=(cfg,), daemon=True)
 80        self._thread.start()
 81
 82    def join(self, timeout: float = 0) -> bool:
 83        if self._thread is None:
 84            return True
 85        self._thread.join(timeout)
 86        return not self._thread.is_alive()

 88    # ------------------------------------------------------------------
 89    def _post(self, event) -> None:
 90        self._q.put(event)
 91
 92    def _run(self, cfg) -> None:
 93        try:
 94            self._pipeline(cfg)
 95        except Exception as exc:
 96            self._post(Error(exc, traceback.format_exc()))

 99    def _build_hooks(self, cfg, margin, board):
100        """Build a `pipeline.PipelineHooks` that posts throttled GUI events.
101
102        The shared `run_placement` / `run_routing` call these at each progress
103        point; here they translate into `Phase` / `Progress` / `BoardSnap` events.
104        Snapshot copies are taken from the live ``board`` at a consistent instant.
105
106        Args:
107            cfg: the run configuration (for budgets shown in the progress events).
108            margin: placement preview margin (mm), passed to `_snap_pads`.
109            board: the live board the pipeline mutates (snapshotted for the canvas).
110
111        Returns:
112            A `pyautoroute.pipeline.PipelineHooks`.
113        """
114        from pyautoroute import pipeline
115
116        post = self._post
117        cancel = self._cancel
118        n_fps = len(board.footprints)
119        # mutable throttle / timer state shared across the callbacks
120        st = {"tag": "", "place_t0": time.monotonic(), "anneal_t0": time.monotonic(),
121              "prog": 0.0, "psnap_t": 0.0, "psnap_best": float("inf"),
122              "route_prog": 0.0, "route_snap": 0.0, "anneal_prog": 0.0}
123
124        def phase(name):
125            if name.startswith("annealing"):
126                st["anneal_t0"] = time.monotonic()
127            post(Phase(f"{st['tag']}{name}"))
128
129        def place_run(k, n):
130            st["tag"] = f"run {k + 1}/{n} " if n > 1 else ""
131            st["place_t0"] = time.monotonic()
132            if k > 0:                                   # later runs get a new phase
133                post(Phase(f"{st['tag']}placing {n_fps} footprints"))
134
135        def place_progress(it, total, energy, best, temp, accept, ob):
136            if cancel.is_set():
137                return
138            now = time.monotonic()
139            if now - st["prog"] >= _PROGRESS_INTERVAL:
140                st["prog"] = now
141                post(Progress("placing", it, total, energy, best, temp, accept,
142                              0, 0, elapsed=now - st["place_t0"],
143                              budget=cfg.place_time or 0.0))
144            new_best = best < st["psnap_best"]
145            if now - st["psnap_t"] >= (0.5 if new_best else 2.0):
146                if new_best:
147                    st["psnap_best"] = best
148                st["psnap_t"] = now
149                post(BoardSnap(_snap_pads(board, margin),
150                               kind="best" if new_best else "current"))
151                if new_best:
152                    post(BoardSnap(_snap_pads(board, margin), kind="overall_best"))
153
154        def placed(_board):
155            post(BoardSnap(_snap_pads(board, margin)))
156
157        def route_run(k, n):
158            st["tag"] = f"run {k + 1}/{n} " if n > 1 else ""
159
160        def route_progress(done, total, routed, unrouted):
161            now = time.monotonic()
162            if now - st["route_prog"] >= _PROGRESS_INTERVAL:
163                st["route_prog"] = now
164                post(Progress("routing", done, total, 0.0, 0.0, 0.0, 0.0,
165                              routed, unrouted))
166
167        def route_partial(b, g, partial):
168            now = time.monotonic()
169            if now - st["route_snap"] >= _ROUTE_SNAP_INTERVAL:
170                st["route_snap"] = now
171                post(BoardSnap(dataclasses.replace(b), results=list(partial),
172                               grid=g))
173
174        def anneal_progress(it, total, r, u, energy, best, temp, accept, ob):
175            now = time.monotonic()
176            if now - st["anneal_prog"] >= _PROGRESS_INTERVAL:
177                st["anneal_prog"] = now
178                post(Progress("annealing", it, total, energy, best, temp, accept,
179                              r, u, elapsed=now - st["anneal_t0"],
180                              budget=cfg.time_budget or 0.0))
181
182        def anneal_snapshot(b, g, results, k, n):
183            post(BoardSnap(dataclasses.replace(b), results=results, grid=g))
184
185        def anneal_best(b, g, results):
186            post(BoardSnap(dataclasses.replace(b), results=results, grid=g,
187                           kind="best"))
188
189        def overall_best(b, g, results, energy):
190            post(BoardSnap(dataclasses.replace(b), results=list(results), grid=g,
191                           kind="overall_best"))
192
193        return pipeline.PipelineHooks(
194            phase=phase, place_run=place_run, place_progress=place_progress,
195            placed=placed, route_run=route_run, route_progress=route_progress,
196            route_partial=route_partial, anneal_progress=anneal_progress,
197            anneal_snapshot=anneal_snapshot, anneal_best=anneal_best,
198            overall_best=overall_best)

200    def _pipeline(self, cfg) -> None:
201        from pyautoroute import __version__, geometry, pcb, pipeline
202        from pyautoroute import placement as place_mod
203        from pyautoroute import router
204        from pyautoroute.autoroute import (
205            _results_to_nodes, default_output, default_pitch,
206            default_place_buffer,
207        )
208        from pyautoroute.rules import load_rules
209
210        input_path = Path(cfg.input)
211        place = cfg.place
212        place_only = cfg.place_only
213        out_path = default_output(input_path, place=place, place_only=place_only)
214
215        self._post(Phase("parsing board + rules"))
216        board = pcb.load_board(input_path)
217        if getattr(cfg, "fix_values", False):
218            pcb.fix_value_layers(board)
219
220        fill_nets = pcb.zone_fill_nets(board)
221        if fill_nets:
222            current_excludes = list(cfg.exclude_net or [])
223            for n in sorted(fill_nets):
224                if n not in current_excludes:
225                    current_excludes.append(n)
226            cfg.exclude_net = current_excludes
227
228        pro_path = (Path(cfg.pro) if cfg.pro
229                    else input_path.with_suffix(".kicad_pro"))
230        if not pro_path.exists():
231            pro_path = input_path.with_name(input_path.stem + ".kicad_pro")
232        rules = load_rules(pro_path)
233        pitch = cfg.grid if cfg.grid else default_pitch(rules)
234
235        margin = cfg.place_margin
236        hooks = self._build_hooks(cfg, margin, board)
237
238        # ---- placement -----------------------------------------------
239        if place or place_only:
240            buf = (default_place_buffer(rules) if cfg.place_buffer is None
241                   else cfg.place_buffer)
242            pp = place_mod.PlaceParams(
243                iters=cfg.place_iters, time_budget=cfg.place_time, seed=cfg.seed,
244                exclude=cfg.exclude_net or [],
245                overlap_weight=cfg.place_overlap_weight,
246                compact_weight=cfg.place_compact_weight,
247                edge_weight=cfg.place_edge_weight,
248                keep_outline=getattr(cfg, "keep_outline", False), buffer=buf,
249                t_start=cfg.place_temps[0], t_end=cfg.place_temps[1],
250                step=cfg.place_step, rotate_mode=cfg.place_rotate)
251            pipeline.run_placement(board, place_params=pp,
252                                   place_runs=max(1, cfg.place_runs),
253                                   seed=cfg.seed, place_margin=margin,
254                                   hooks=hooks, cancel=self._cancel)
255
256        if place_only:
257            self._post(Phase("writing placed board"))
258            pcb.stamp_comment(board,
259                f"PyAutoRoute v{__version__} — placed "
260                f"{datetime.date.today().isoformat()}")
261            pcb.write_board(board, out_path, new_nodes=None, strip_free_vias=True)
262            if fill_nets:
263                pcb.try_refill_zones(out_path)
264            placed = pcb.load_board(out_path)
265            violations = geometry.clearance_violations(placed, rules)
266            self._post(Done(str(out_path), 0, 0, 0, 0.0, 0, violations, placed))
267            return
268
269        if self._cancel.is_set():
270            self._post(Phase("cancelled"))
271            return
272
273        # ---- routing -------------------------------------------------
274        annealing = bool(cfg.iters or cfg.time_budget)
275        runs = max(1, cfg.runs)
276        if runs > 1 and not annealing:
277            runs = 1                                    # greedy is deterministic
278        route_params = router.RouteParams(via_cost=cfg.via_weight)
279        route_kw = dict(annealing=annealing, iters=cfg.iters,
280                        time_budget=cfg.time_budget,
281                        unrouted_weight=cfg.unrouted_weight,
282                        anneal_temps=cfg.anneal_temps, via_weight=cfg.via_weight)
283        res = pipeline.run_routing(
284            board, rules, pitch, route_params=route_params, route_kw=route_kw,
285            seed=cfg.seed, runs=runs, jobs=1, snapshots=_ANNEAL_SNAP_COUNT,
286            exclude=cfg.exclude_net or [], hooks=hooks, cancel=self._cancel)
287
288        if res.results is None:
289            self._post(Phase("cancelled before routing completed"))
290            return
291
292        grid = res.grid
293        final_results = res.results
294        routed, unrouted = res.routed, res.unrouted
295        length, vias = res.length, res.vias
296
297        self._post(Phase("writing routed board"))
298        _mode = "placed + routed" if place else "routed"
299        pcb.stamp_comment(board,
300            f"PyAutoRoute v{__version__} — {_mode} "
301            f"{datetime.date.today().isoformat()}")
302        pcb.write_board(board, out_path,
303                        new_nodes=_results_to_nodes(board, grid, final_results),
304                        strip_free_vias=True)
305        if fill_nets:
306            pcb.try_refill_zones(out_path)
307        routed_board = pcb.load_board(out_path)
308        violations = geometry.clearance_violations(routed_board, rules)
309        total = routed + unrouted
310        self._post(BoardSnap(routed_board))
311        self._post(Done(str(out_path), total, routed, unrouted,
312                        length, vias, violations, routed_board))
