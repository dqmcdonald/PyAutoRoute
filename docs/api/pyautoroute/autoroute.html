pyautoroute.autoroute

Command-line entry point: parse a board, route it, write a routed copy.

python -m pyautoroute.autoroute INPUT.kicad_pcb [options]

Orchestrates parse -> grid build -> route -> write, prints a live text progress display (unless --quiet), reports metrics, and runs an in-repo clearance self-check on the result.

  37class Reporter:
  38    """Live single-line progress display on a TTY; quiet/plain otherwise.
  39
  40    If ``log_path`` is given, every phase and a (throttled) trace of routing /
  41    annealing progress is also appended to that file, regardless of ``quiet``,
  42    alongside whatever ``log()`` is called with directly (parameter dump, final
  43    metrics). The log is plain text, one timestamped line per entry.
  44    """
  45
  46    def __init__(self, stream=sys.stderr, quiet: bool = False,
  47                 log_path: Path | None = None):
  48        self.stream = stream
  49        self.quiet = quiet
  50        self.tty = (not quiet) and hasattr(stream, "isatty") and stream.isatty()
  51        self._t0 = time.time()
  52        self._c0 = time.process_time()
  53        self.log_file = open(log_path, "w") if log_path else None
  54        self.tag = ""
  55
  56    def phase(self, name: str) -> None:
  57        """Announce a new phase on the display and in the log.
  58
  59        Args:
  60            name: the phase name, e.g. ``"routing 94 connections"``.
  61        """
  62        self.log(f"{name} ...")
  63        if self.quiet:
  64            return
  65        self._write(f"[{self._elapsed():6.1f}s] {name} ...")
  66        if not self.tty:
  67            self.stream.write("\n")
  68
  69    def routing(self, done: int, total: int, routed: int, unrouted: int) -> None:
  70        """Report greedy-routing progress.
  71
  72        Args:
  73            done: connections attempted so far.
  74            total: total connections to route.
  75            routed: count routed successfully.
  76            unrouted: count that failed.
  77        """
  78        msg = (f"{self.tag}routing {done}/{total}  routed={routed} failed={unrouted}")
  79        if done == total or done % 10 == 0:
  80            self.log(msg)
  81        if self.quiet:
  82            return
  83        line = f"[{self._elapsed():6.1f}s] {msg}"
  84        if self.tty:
  85            self._write(line)
  86        elif done == total or done % 10 == 0:
  87            self.stream.write(line + "\n")
  88
  89    def annealing(self, it, total, routed, unrouted, energy, best, temp,
  90                  accept, *, elapsed: float = 0.0, budget: float = 0.0,
  91                  overall_best: float | None = None) -> None:
  92        """Report an annealing iteration.
  93
  94        Args:
  95            it: iterations completed.
  96            total: nominal iteration count (for the progress fraction).
  97            routed: connections currently routed.
  98            unrouted: connections currently unrouted.
  99            energy: current energy.
 100            best: best energy seen so far.
 101            temp: current annealing temperature.
 102            accept: fraction of recent moves accepted (0..1); falls as T cools.
 103            elapsed: seconds elapsed in this run (non-zero enables time display).
 104            budget: time budget in seconds.
 105            overall_best: best energy across all runs so far (shown when > 1 run).
 106        """
 107        iter_str = (f"{max(0.0, budget - elapsed):.0f}s rem"
 108                    if budget > 0 else f"{it}/{total}")
 109        ob_str = (f"  ob={overall_best:7.1f}" if overall_best is not None else "")
 110        msg = (f"{self.tag}anneal {iter_str}  T={temp:5.2f}  E={energy:7.1f}  "
 111               f"best={best:7.1f}{ob_str}  acc={accept*100:3.0f}%  "
 112               f"routed={routed} failed={unrouted}")
 113        if it % 25 == 0:
 114            self.log(msg)
 115        if self.quiet:
 116            return
 117        line = f"[{self._elapsed():6.1f}s] {msg}"
 118        if self.tty:
 119            self._write(line)
 120        elif it % 25 == 0:
 121            self.stream.write(line + "\n")

 123    def placing(self, it, total, energy, best, temp, accept,
 124               *, elapsed: float = 0.0, budget: float = 0.0,
 125               overall_best: float | None = None) -> None:
 126        """Report a placement-annealing iteration.
 127
 128        Args:
 129            it: iterations completed.
 130            total: nominal iteration count (for the progress fraction).
 131            energy: current placement energy.
 132            best: best energy seen so far.
 133            temp: current annealing temperature.
 134            accept: fraction of recent moves accepted (0..1); falls as T cools.
 135            elapsed: seconds elapsed in this run (non-zero enables time display).
 136            budget: time budget in seconds.
 137            overall_best: best energy across all runs so far (shown when > 1 run).
 138        """
 139        iter_str = (f"{max(0.0, budget - elapsed):.0f}s rem"
 140                    if budget > 0 else f"{it}/{total}")
 141        ob_str = (f"  ob={overall_best:8.1f}" if overall_best is not None else "")
 142        msg = (f"{self.tag}place {iter_str}  T={temp:5.2f}  E={energy:8.1f}  "
 143               f"best={best:8.1f}{ob_str}  acc={accept*100:3.0f}%")
 144        if it % 25 == 0:
 145            self.log(msg)
 146        if self.quiet:
 147            return
 148        line = f"[{self._elapsed():6.1f}s] {msg}"
 149        if self.tty:
 150            self._write(line)
 151        elif it % 25 == 0:
 152            self.stream.write(line + "\n")

 154    def log(self, msg: str) -> None:
 155        """Append a timestamped line to the log file (no-op without one)."""
 156        if self.log_file is not None:
 157            self.log_file.write(f"[{self._elapsed():8.2f}s] {msg}\n")
 158            self.log_file.flush()
 159
 160    def close(self) -> None:
 161        """Close the log file, if one is open."""
 162        if self.log_file is not None:
 163            self.log_file.close()
 164            self.log_file = None
 165
 166    def done(self) -> None:
 167        """Terminate the live TTY progress line with a newline."""
 168        if self.tty:
 169            self.stream.write("\n")
 170            self.stream.flush()
 171
 172    def _elapsed(self) -> float:
 173        """Seconds elapsed since this reporter was created."""
 174        return time.time() - self._t0
 175
 176    def runtime(self) -> tuple[float, float]:
 177        """Wall-clock and process CPU seconds elapsed since this reporter began.
 178
 179        Returns:
 180            ``(real, cpu)`` — elapsed wall-clock time and total (user+system)
 181            CPU time of this process, both in seconds.
 182        """
 183        return time.time() - self._t0, time.process_time() - self._c0
 184
 185    def _write(self, msg: str) -> None:
 186        """Write a formatted line to the display.
 187
 188        Args:
 189            msg: the already-formatted line; emitted in place (``\\r``) on a TTY
 190                or as its own line otherwise.
 191        """
 192        if self.tty:
 193            self.stream.write("\r\033[K" + msg)
 194            self.stream.flush()
 195        else:
 196            self.stream.write(msg + "\n")

 199def default_output(input_path: Path, place: bool = False,
 200                   place_only: bool = False) -> Path:
 201    """Default output path beside the input, named for what the run produced.
 202
 203    Args:

 230def default_pitch(rules) -> float:
 231    """Default grid pitch derived from the rules' Default class.
 232
 233    Args:
 234        rules: the `pyautoroute.rules.DesignRules`.
 235
 236    Returns:
 237        ``track_width/2 + clearance`` (mm), rounded to 4 places.

 239    dc = rules.default_class

 243def default_place_buffer(rules) -> float:
 244    """Default placement keep-out gap (mm) derived from the design rules.
 245
 246    The placement pass keeps footprints at least this far apart; a value derived
 247    from the clearance leaves room for routing between adjacent parts so the
 248    placed board does not fail DRC.
 249
 250    Args:
 251        rules: the `pyautoroute.rules.DesignRules`.
 252
 253    Returns:
 254        ``max(2 x max-class-clearance, 0.5)`` mm, rounded to 4 places.
 255    """
 256    max_clear = max([c.clearance for c in rules.classes.values()]
 257                    + [rules.min_clearance])
 258    return round(max(2.0 * max_clear, 0.5), 4)

 261# A grid coarser than this multiple of the rules-derived pitch often can't place
 262# a node in the tight gap beside a pad, forcing vias where a finer grid would
 263# route on one layer. Empirically a pad-flanking single-layer route survives at
 264# ~2x the natural pitch but is lost beyond it.
 265COARSE_GRID_FACTOR = 2.0

 268def coarse_grid_note(pitch: float, natural: float) -> str | None:
 269    """Warn when the grid pitch is too coarse relative to the design rules.
 270
 271    A grid much coarser than the rules-derived pitch (``track/2 + clearance``)
 272    often cannot fit a node in the clearance gap beside a pad, so the router is
 273    forced to via under it where a finer grid would route on a single layer.
 274
 275    Args:
 276        pitch: the routing grid pitch actually in use (mm).
 277        natural: the rules-derived pitch (`default_pitch`), in mm.
 278
 279    Returns:
 280        A warning string when `pitch` exceeds ``COARSE_GRID_FACTOR x natural``,
 281        else `None`.
 282    """
 283    if natural <= 0 or pitch <= COARSE_GRID_FACTOR * natural:
 284        return None
 285    return (f"grid pitch {pitch:g} mm is {pitch / natural:.1f}x the rules-derived "
 286            f"pitch ({natural:g} mm); a coarse grid can force vias where a finer "
 287            f"grid would route on one layer — consider --grid {natural:g} "
 288            f"(or omit --grid)")

 394def _place_params_from_args(args, board, rules, rep):
 395    """Build `placement.PlaceParams` from the CLI args.
 396
 397    Resolves the placement-buffer default (from the design rules) and the
 398    ``--keep-outline`` fallback — a synthesised or absent Edge.Cuts can't be kept,
 399    so the flag is dropped with a note. Shared by the single-pass placement in
 400    `run` and the per-cycle placement in `_run_cycles`.
 401
 402    Args:
 403        args: the parsed CLI namespace.
 404        board: the parsed board (for the keep-outline Edge.Cuts check).
 405        rules: the design rules (for the buffer default).
 406        rep: the `Reporter` (for the fallback note).
 407
 408    Returns:
 409        ``(params, keep_outline)`` — the placement params and the resolved
 410        keep-outline flag.
 411    """
 412    if args.place_buffer is None:
 413        args.place_buffer = default_place_buffer(rules)
 414    keep_outline = bool(getattr(args, "keep_outline", False))
 415    if keep_outline and (not board.outline or board.outline_synthesized):
 416        keep_outline = False
 417        msg = ("--keep-outline ignored: the board has no Edge.Cuts to keep; "
 418               "regenerating a bounding outline instead")
 419        rep.log(msg)
 420        if not args.quiet:
 421            print(f"  note: {msg}")
 422    pp = placement.PlaceParams(
 423        iters=args.place_iters, time_budget=args.place_time, seed=args.seed,
 424        exclude=args.exclude_net, overlap_weight=args.place_overlap_weight,
 425        compact_weight=args.place_compact_weight, buffer=args.place_buffer,
 426        edge_weight=args.place_edge_weight, keep_outline=keep_outline,
 427        t_start=args.place_temps[0], t_end=args.place_temps[1],
 428        step=args.place_step, rotate_mode=args.place_rotate)
 429    return pp, keep_outline

 432def run(args: argparse.Namespace, _print_version: bool = True) -> int:
 433    """Execute the full pipeline: parse -> grid -> route -> (anneal) -> write.
 434
 435    Writes the routed board, optional snapshots and log, runs the clearance
 436    self-check, and prints the metrics report (including the run's wall-clock
 437    and CPU time, also mirrored to the log).
 438
 439    Args:
 440        args: the parsed CLI namespace (see `build_parser`).
 441        _print_version: if False, suppress the opening ``PyAutoRoute vX.Y``
 442            line (used by `main` when the settings header already printed it).
 443
 444    Returns:
 445        Process exit code: 0 if the self-check is clean, 2 if it finds a
 446        clearance violation.
 447    """
 448    input_path = Path(args.input)
 449    out_path = (Path(args.output) if args.output
 450                else default_output(input_path, place=args.place,
 451                                    place_only=args.place_only))
 452    pro_path = Path(args.pro) if args.pro else default_pro(input_path)
 453    rep = Reporter(quiet=args.quiet, log_path=_resolve_log_path(args, out_path))
 454    if _print_version:
 455        print(f"PyAutoRoute {__version__}")
 456
 457    rep.phase("parsing board + rules")
 458    board = pcb.load_board(input_path)
 459    if board.outline_synthesized:
 460        print("  note: no Edge.Cuts outline found — default bounding-box outline added")
 461
 462    fill_nets = pcb.zone_fill_nets(board)
 463    if fill_nets:
 464        current_excludes = list(args.exclude_net or [])
 465        for n in sorted(fill_nets):
 466            if n not in current_excludes:
 467                current_excludes.append(n)
 468                print(f"  fill zone:     auto-excluding net '{n}' (has copper pour)")
 469        args.exclude_net = current_excludes
 470
 471    rules = load_rules(pro_path)
 472    pitch = args.grid if args.grid else default_pitch(rules)
 473
 474    if getattr(args, "fix_values", False):
 475        n = pcb.fix_value_layers(board)
 476        if n:
 477            print(f"  fix-values:    moved {n} Value text node(s) to silkscreen")
 478
 479    if board.segments:
 480        from pyautoroute.report import routing_stats
 481        init = routing_stats(board, rules)
 482        print(f"  initial board: {init.summary()}")
 483
 484    cycles = max(1, getattr(args, "cycles", 1))
 485    if cycles > 1 and not args.place:
 486        print("  note: --cycles needs --place (it selects placements by how they "
 487              "route); ignoring")
 488        cycles = 1
 489    if cycles > 1 and args.place_only:
 490        print("  note: --cycles has no effect with --place-only (no routing); "
 491              "ignoring")
 492        cycles = 1
 493    if getattr(args, "place_feedback", False) and cycles <= 1:
 494        print("  note: --place-feedback needs --cycles > 1 (it learns from each "
 495              "cycle's routing); ignoring")
 496        args.place_feedback = False
 497    if cycles > 1:
 498        return _run_cycles(args, rep, input_path, out_path, rules, pitch,
 499                           board, fill_nets, cycles)
 500
 501    if args.place or args.place_only:
 502        pp, keep_outline = _place_params_from_args(args, board, rules, rep)
 503        place_runs = max(1, args.place_runs)
 504        n_fps = len(board.footprints)
 505        _place_t0 = [time.monotonic()]
 506
 507        def _pl_run(k, n):
 508            rep.tag = f"run {k + 1}/{n}: " if n > 1 else ""
 509            _place_t0[0] = time.monotonic()
 510            if k > 0:
 511                rep.phase(f"{rep.tag}placing {n_fps} footprints (annealing)")
 512
 513        def _pl_progress(it, total, energy, best, temp, accept, ob):
 514            rep.placing(it, total, energy, best, temp, accept,
 515                        elapsed=time.monotonic() - _place_t0[0],
 516                        budget=args.place_time or 0.0, overall_best=ob)
 517
 518        place_hooks = PipelineHooks(
 519            phase=lambda name: rep.phase(f"{rep.tag}{name}"),
 520            place_run=_pl_run, place_progress=_pl_progress)
 521        pout = run_placement(board, place_params=pp, place_runs=place_runs,
 522                             seed=args.seed, place_margin=args.place_margin,
 523                             hooks=place_hooks)
 524        rep.tag = ""
 525        rep.done()
 526        if place_runs > 1:
 527            rep.log(f"best of {place_runs} placement runs: energy {pout.best_energy:.1f}")
 528        summary = (f"place: {pout.iterations} iters, "
 529                   f"{pout.accepted} accepted ({pout.accept_ratio*100:.0f}%), "
 530                   f"{pout.moved} moved, energy "
 531                   f"{pout.start_energy:.1f} -> {pout.best_energy:.1f}"
 532                   + (f"  (best of {place_runs})" if place_runs > 1 else ""))
 533        breakdown = (f"placement: ratsnest {pout.final_ratsnest:.1f} mm, "
 534                     f"overlap {pout.final_overlap:.1f} mm2, "
 535                     f"bbox {pout.final_bbox:.0f} mm2")
 536        if any(fp.edge_affinity for fp in board.footprints):
 537            breakdown += f", edge {pout.final_edge:.1f} mm"
 538        rep.log(summary)
 539        rep.log(breakdown)
 540        if not args.quiet:
 541            print(f"\n  {summary}")
 542            print(f"  {breakdown}")
 543
 544    if args.place_only:
 545        rep.phase("writing placed board")
 546        _stamp(board, "placed")
 547        pcb.write_board(board, out_path, new_nodes=None, strip_free_vias=True)
 548        if fill_nets:
 549            ok = pcb.try_refill_zones(out_path)
 550            if ok:
 551                rep.log("zones refilled via kicad-cli")
 552                print("  zones:         copper fill refilled (kicad-cli)")
 553            else:
 554                rep.log("zone refill skipped — kicad-cli not found or failed")
 555                print("  note: kicad-cli not available; open in KiCad to refill copper zones")
 556        placed_board = pcb.load_board(out_path)
 557        violations = geometry.clearance_violations(placed_board, rules)
 558        rep.done()
 559        _report_placed(rep, out_path, board, violations)
 560        return _finish(rep, args, out_path, placed_board, violations)
 561
 562    if args.auto:
 563        from . import tune
 564        rep.phase("auto: probing grid/via settings")
 565        scored = tune.sweep(board, rules,
 566                            tune.default_grid(time_budget=args.auto_probe_time),
 567                            seeds=(args.seed,), unrouted_weight=args.unrouted_weight,
 568                            via_weight=args.via_weight)
 569        best = tune.best_config(scored)
 570        chosen_pitch = round(default_pitch(rules) * best.grid_mult, 4)
 571        rep.done()
 572        bm = scored[0].metrics[0]
 573        total = bm.routed + bm.unrouted
 574        chosen = (f"auto: best probe grid={chosen_pitch} mm (x{best.grid_mult}), "
 575                  f"via-weight={best.via_weight} -> {bm.routed}/{total} routed, "
 576                  f"{bm.length:.0f} mm, {bm.vias} vias")
 577        print(f"\n  {chosen}")
 578        rep.log(chosen)
 579        apply_auto = True
 580        if sys.stdin.isatty() and not args.auto_yes:
 581            apply_auto = input("  apply these settings? [Y/n] ").strip().lower() \
 582                in ("", "y", "yes")
 583        if apply_auto:
 584            args.grid, args.via_weight, pitch = chosen_pitch, best.via_weight, chosen_pitch
 585        else:
 586            print("  auto: keeping the given settings")
 587
 588    rep.phase("building netlist (MST rats-nest)")
 589    conns = netlist.build_connections(board, exclude=args.exclude_net)
 590    excluded = sorted({p.net for p in board.pads if p.net
 591                       and netlist.is_excluded(p.net, args.exclude_net)})
 592
 593    rep.phase(f"building {pitch}mm routing grid")
 594    grid = Grid(board, rules, pitch)
 595
 596    runs = max(1, args.runs)
 597    if runs > 1 and not (args.iters or args.time_budget):
 598        print("  note: --runs > 1 has no effect without --iters/--time "
 599              "(greedy routing is deterministic); using 1 run")
 600        runs = 1
 601    # Worker count for parallel best-of-N: --jobs 0 means "use as many workers
 602    # as runs, capped at the CPU count". --jobs 1 (the default) keeps the
 603    # byte-identical sequential path. Never spawn more workers than runs.
 604    jobs = args.jobs if args.jobs and args.jobs > 0 else (os.cpu_count() or 1)
 605    jobs = max(1, min(jobs, runs))
 606    parallel = runs > 1 and jobs > 1
 607    # snapshots only make sense during a single annealing run
 608    snap_n = args.snapshots
 609    if snap_n and not (args.iters or args.time_budget):
 610        print("  note: --snapshots needs --iters or --time (annealing); ignoring")
 611        snap_n = 0
 612    if snap_n and runs > 1:
 613        print("  note: --snapshots needs a single run; ignoring with --runs > 1")
 614        snap_n = 0
 615    snap_dir = out_path.parent / "snapshots" if snap_n else None
 616    if snap_dir is not None:
 617        snap_dir.mkdir(parents=True, exist_ok=True)

 620        sp = snap_dir / f"{input_path.stem}_anneal_{k:02d}of{n:02d}.kicad_pcb"
 621        pcb.write_board(board, sp, new_nodes=_results_to_nodes(board, grid, results),
 622                        strip_free_vias=True)
 623        nrouted = sum(1 for r in results if r is not None)
 624        rep.log(f"snapshot {k}/{n} -> {sp}  routed={nrouted}/{len(results)}")
 625
 626    _log_params(rep, args, input_path, out_path, pro_path, pitch,
 627                board, conns, grid, snap_n)
 628
 629    note = coarse_grid_note(pitch, default_pitch(rules))
 630    if note:
 631        print(f"  warning: {note}")
 632        rep.log(f"warning: {note}")

 635                                search_margin=args.search_margin)
 636    order = netlist.greedy_order(conns)
 637    annealing = bool(args.iters or args.time_budget)
 638    route_kw = dict(annealing=annealing, iters=args.iters,
 639                    time_budget=args.time_budget,
 640                    unrouted_weight=args.unrouted_weight,
 641                    anneal_temps=args.anneal_temps, via_weight=args.via_weight)
 642    _anneal_t0 = [0.0]
 643
 644    def _rt_phase(name):
 645        rep.done()
 646        if name.startswith("annealing"):
 647            _anneal_t0[0] = time.monotonic()
 648        rep.phase(f"{rep.tag}{name}")
 649
 650    def _rt_anneal(it, total, r, u, energy, best, temp, accept, ob):
 651        rep.annealing(it, total, r, u, energy, best, temp, accept,
 652                      elapsed=time.monotonic() - _anneal_t0[0],
 653                      budget=args.time_budget or 0.0, overall_best=ob)
 654
 655    def _rt_run_done(k, n, energy, summary, metrics):
 656        if parallel:                                   # completion-ordered logging
 657            rep.log(f"run {k + 1}/{n} done: energy {energy:.1f}"
 658                    + (f"  ({summary})" if summary else ""))
 659        elif summary:                                  # sequential annealing summary
 660            line = f"{rep.tag}{summary}"
 661            rep.log(line)
 662            if not args.quiet:
 663                print(f"\n  {line}")
 664        elif n > 1:
 665            rep.log(f"{rep.tag}energy {energy:.1f}")
 666
 667    route_hooks = PipelineHooks(
 668        phase=_rt_phase,
 669        route_run=lambda k, n: setattr(
 670            rep, "tag", f"run {k + 1}/{n}: " if n > 1 else ""),
 671        route_progress=rep.routing,
 672        anneal_progress=_rt_anneal,
 673        anneal_snapshot=((lambda b, g, res, k, n: on_snapshot(k, n, res))
 674                         if snap_n else None),
 675        route_run_done=_rt_run_done)
 676
 677    res = run_routing(board, rules, pitch, route_params=params, route_kw=route_kw,
 678                      seed=args.seed, runs=runs, jobs=jobs, snapshots=snap_n,
 679                      exclude=args.exclude_net, grid=grid, conns=conns, order=order,
 680                      hooks=route_hooks)
 681    rep.done()
 682    rep.tag = ""
 683    final_results = (res.results if res.results is not None
 684                     else [None] * len(conns))
 685    routed, unrouted, length, vias = res.routed, res.unrouted, res.length, res.vias
 686    best_energy = res.energy
 687
 688    if runs > 1:
 689        best_line = f"best of {runs} runs: energy {best_energy:.1f}"
 690        rep.log(best_line)
 691        if not args.quiet:
 692            print(f"\n  {best_line}")
 693    if snap_n:
 694        print(f"  snapshots:     {snap_n} written to {snap_dir}/")
 695
 696    rep.phase("writing routed board")
 697    mode = "placed + routed" if args.place else "routed"
 698    _stamp(board, mode)
 699    pcb.write_board(board, out_path,
 700                    new_nodes=_results_to_nodes(board, grid, final_results),
 701                    strip_free_vias=True)
 702
 703    if fill_nets:
 704        ok = pcb.try_refill_zones(out_path)
 705        if ok:
 706            rep.log("zones refilled via kicad-cli")
 707            print("  zones:         copper fill refilled (kicad-cli)")
 708        else:
 709            rep.log("zone refill skipped — kicad-cli not found or failed")
 710            print("  note: kicad-cli not available; open in KiCad to refill copper zones")
 711
 712    # reload the written board and self-check clearances
 713    routed_board = pcb.load_board(out_path)
 714    violations = geometry.clearance_violations(routed_board, rules)
 715    rep.done()
 716
 717    _report(rep, out_path, len(conns), routed, unrouted, length, vias,
 718            violations, excluded)
 719    return _finish(rep, args, out_path, routed_board, violations)

 722def _run_cycles(args, rep, input_path, out_path, rules, pitch, board, fill_nets,
 723                cycles) -> int:
 724    """Best-of-cycles: keep the best-*routing* of N independent place+route runs.
 725
 726    Each cycle re-loads the board and runs one placement + one routing through
 727    `pipeline.run_cycle` (seed ``args.seed + k``), so cycles are independent and
 728    selected on the true routed objective — fewest unrouted, then lowest energy —
 729    rather than on placement energy. Cycles run sequentially with live progress,
 730    or across ``--jobs`` worker processes (progress suppressed, one line per
 731    finished cycle). The winning cycle's board is written, zone-refilled,
 732    self-checked and reported exactly as `run` does.
 733
 734    Args:
 735        args: the parsed CLI namespace.
 736        rep: the reporter.
 737        input_path: the source board path (re-read each cycle).
 738        out_path: the output board path.
 739        rules: the design rules.
 740        pitch: routing-grid pitch (mm).
 741        board: the already-parsed board (for param resolution + exclude reporting).
 742        fill_nets: nets with copper pours (for the zone refill).
 743        cycles: number of cycles (> 1).
 744
 745    Returns:
 746        Process exit code (0 clean, 2 on a clearance violation).
 747    """
 748    pp, _keep = _place_params_from_args(args, board, rules, rep)
 749    route_params = router.RouteParams(via_cost=args.via_weight,
 750                                      search_margin=args.search_margin)
 751    route_kw = dict(annealing=bool(args.iters or args.time_budget),
 752                    iters=args.iters, time_budget=args.time_budget,
 753                    unrouted_weight=args.unrouted_weight,
 754                    anneal_temps=args.anneal_temps, via_weight=args.via_weight)
 755    base_seed = args.seed
 756    jobs = args.jobs if args.jobs and args.jobs > 0 else (os.cpu_count() or 1)
 757    jobs = max(1, min(jobs, cycles))
 758    # Congestion feedback couples each cycle to the previous one's routing, so it
 759    # cannot run across independent workers — it forces the sequential path.
 760    feedback = bool(getattr(args, "place_feedback", False))
 761    parallel = jobs > 1 and not feedback
 762
 763    def _cycle_line(k, cr):
 764        return (f"cycle {k}/{cycles}: routed {cr.routed}/{cr.n_conns}, "
 765                f"energy {cr.energy:.1f}, {cr.vias} vias"
 766                + (f", {cr.unrouted} unrouted" if cr.unrouted else ""))
 767
 768    results: list = []
 769    if parallel:
 770        rep.phase(f"best-of-cycles: {cycles} place+route cycles "
 771                  f"across {jobs} workers")
 772        payloads = [(str(input_path), rules, pitch, pp, route_params,
 773                     route_kw, args.place_margin, base_seed + k)
 774                    for k in range(cycles)]
 775        try:
 776            with concurrent.futures.ProcessPoolExecutor(max_workers=jobs) as ex:
 777                futs = [ex.submit(_cycle_worker, p) for p in payloads]
 778                for fut in concurrent.futures.as_completed(futs):
 779                    cr = fut.result()
 780                    results.append(cr)
 781                    line = _cycle_line(len(results), cr)
 782                    rep.log(line)
 783                    if not args.quiet:
 784                        print(f"  {line}")
 785        except KeyboardInterrupt:
 786            rep.log("interrupted — keeping best cycle so far")
 787            if not results:
 788                raise
 789        rep.done()
 790    else:
 791        if feedback:
 792            rep.phase(f"best-of-cycles: {cycles} place+route cycles with "
 793                      f"congestion feedback (weight {args.congestion_weight:g})")
 794        # Congestion feedback (--place-feedback): a fixed board-wide field frame is
 795        # built once, then each cycle (after cycle 0) re-places under the field
 796        # accumulated from the previous cycles' routing, decayed so signal builds
 797        # without one cycle dominating. The best-routing cycle is still selected,
 798        # so feedback can only help or be discarded.
 799        frame = router.congestion_frame(board, pitch) if feedback else None
 800        field = None
 801        for k in range(cycles):
 802            rep.tag = f"cycle {k + 1}/{cycles}: "
 803            hooks = CycleHooks(
 804                phase_cb=lambda n: rep.phase(rep.tag + n),
 805                place_progress=(lambda it, total, e, b, t, acc:
 806                                rep.placing(it, total, e, b, t, acc)),
 807                route_progress=rep.routing,
 808                anneal_progress=(lambda it, total, r, u, e, b, t, acc:
 809                                 rep.annealing(it, total, r, u, e, b, t, acc)),
 810            )
 811            pp_k = pp
 812            if feedback and field is not None:
 813                pp_k = replace(pp, congestion_field=field,
 814                               congestion_weight=args.congestion_weight)
 815            cr = run_cycle(input_path, rules, pitch, pp_k, route_params,
 816                           route_kw=route_kw, place_margin=args.place_margin,
 817                           seed=base_seed + k, hooks=hooks)
 818            rep.done()
 819            results.append(cr)
 820            line = _cycle_line(k + 1, cr)
 821            rep.log(line)
 822            if not args.quiet:
 823                print(f"  {line}")
 824            if feedback and k + 1 < cycles:
 825                new_field = router.congestion_heatmap(cr.conns, cr.results,
 826                                                      cr.grid, frame)
 827                field = (new_field if field is None
 828                         else field.blended(new_field, _FEEDBACK_DECAY))
 829        rep.tag = ""
 830
 831    best = select_best(results)
 832    best_line = (f"best of {cycles} cycles: seed {best.seed} — "
 833                 f"routed {best.routed}/{best.n_conns}, energy {best.energy:.1f}, "
 834                 f"{best.vias} vias"
 835                 + (f", {best.unrouted} unrouted" if best.unrouted else ""))
 836    rep.log(best_line)
 837    if not args.quiet:
 838        print(f"\n  {best_line}")
 839
 840    sel_board, grid, final_results = best.board, best.grid, best.results
 841    excluded = sorted({p.net for p in sel_board.pads if p.net
 842                       and netlist.is_excluded(p.net, args.exclude_net)})
 843
 844    rep.phase("writing placed + routed board")
 845    _stamp(sel_board, "placed + routed")
 846    pcb.write_board(sel_board, out_path,
 847                    new_nodes=_results_to_nodes(sel_board, grid, final_results),
 848                    strip_free_vias=True)
 849
 850    if fill_nets:
 851        ok = pcb.try_refill_zones(out_path)
 852        if ok:
 853            rep.log("zones refilled via kicad-cli")
 854            print("  zones:         copper fill refilled (kicad-cli)")
 855        else:
 856            rep.log("zone refill skipped — kicad-cli not found or failed")
 857            print("  note: kicad-cli not available; open in KiCad to refill copper zones")
 858
 859    routed_board = pcb.load_board(out_path)
 860    violations = geometry.clearance_violations(routed_board, rules)
 861    rep.done()
 862
 863    _report(rep, out_path, best.n_conns, best.routed, best.unrouted, best.length,
 864            best.vias, violations, excluded)
 865    return _finish(rep, args, out_path, routed_board, violations)

 875        args: the parsed CLI namespace.
 876        out_path: the output board path (basis for the plot/log names).

 878        violations: the self-check violations (empty == clean) for the exit code.
 879
 880    Returns:
 881        Process exit code: 0 if `violations` is empty, else 2.
 882    """
 883    if args.debug_plot:
 884        from . import visualize

 895
 896    if rep.log_file is not None and not args.quiet:
 897        print(f"  log:           {_resolve_log_path(args, out_path)}")
 898    rep.close()
 899    return 0 if not violations else 2

 902def _report(rep: Reporter, out_path, n_conns, routed, unrouted, length, vias,
 903            violations, excluded) -> None:
 904    """Print the final metrics summary and mirror it to the log.

 991def _parse_bool(text: str) -> bool:
 992    """Parse a boolean from a settings-file value.
 993
 994    Args:
 995        text: the raw value (e.g. ``"true"``, ``"0"``, ``"yes"``).
 996
 997    Returns:
 998        The boolean.
 999
1000    Raises:
1001        ValueError: if `text` is not a recognised boolean.
1002    """
1003    low = text.strip().lower()
1004    if low in ("true", "1", "yes", "on"):
1005        return True
1006    if low in ("false", "0", "no", "off"):
1007        return False
1008    raise ValueError(f"not a boolean: {text!r}")

1011def _coerce_config_value(action, raw: str):
1012    """Coerce a raw settings-file string to the option's Python type.
1013
1014    Args:
1015        action: the argparse action for the option.
1016        raw: the raw string from the settings file.
1017
1018    Returns:
1019        The typed value (bool for flags, list for append/2-tuple options,
1020        otherwise the action's ``type`` applied to `raw`).
1021    """
1022    if isinstance(action, (argparse._StoreTrueAction, argparse._StoreFalseAction)):
1023        return _parse_bool(raw)
1024    conv = action.type or str
1025    if isinstance(action, argparse._AppendAction):
1026        return [conv(s.strip()) for s in raw.split(",") if s.strip()]
1027    if action.nargs == 2:
1028        parts = [s for s in raw.replace(",", " ").split() if s]
1029        if len(parts) != 2:
1030            raise ValueError(f"expected 2 values, got {raw!r}")
1031        return [conv(p) for p in parts]
1032    return conv(raw)

1035def load_project_config(path: str | Path,
1036                        parser: argparse.ArgumentParser) -> dict:
1037    """Load a project INI file if it has a ``[pyautoroute]`` section.
1038
1039    Like `load_config` but silently returns ``{}`` when the file does not
1040    exist or lacks the ``[pyautoroute]`` section — so a generic ``*.ini``
1041    used by other tools in the project directory is skipped harmlessly.
1042    Bad values still raise via ``parser.error``.

1044    Args:
1045        path: path to the project INI file (e.g. ``myboard.ini``).
1046        parser: the CLI parser.
1047
1048    Returns:

1050    """
1051    path = Path(path)
1052    if not path.exists():

1055    cp.read(path)
1056    if not cp.has_section(_CONFIG_SECTION):
1057        return {}
1058    return load_config(path, parser)

1071    Raises:
1072        SystemExit: via ``parser.error`` if the file is missing, lacks the
1073            ``[pyautoroute]`` section, names an unknown key, or has a bad value.
1074    """
1075    path = Path(path)
1076    if not path.exists():
1077        parser.error(f"--config file not found: {path}")
1078    cp = configparser.ConfigParser()
1079    cp.read(path)
1080    if not cp.has_section(_CONFIG_SECTION):
1081        parser.error(f"--config file has no [{_CONFIG_SECTION}] section: {path}")
1082    actions = _configurable_actions(parser)
1083    out = {}
1084    for key, raw in cp.items(_CONFIG_SECTION):
1085        dest = key.replace("-", "_")
1086        if dest not in actions:
1087            parser.error(f"--config: unknown option {key!r} in {path}")
1088        try:
1089            out[dest] = _coerce_config_value(actions[dest], raw)
1090        except (ValueError, TypeError) as exc:
1091            parser.error(f"--config: bad value for {key!r}: {exc}")
1092    return out

1124def _append_option(lines: list, dest: str, action, value,
1125                   commented: bool) -> None:
1126    """Append one INI entry (comment + key = value) to *lines*.
1127
1128    If *commented* is True the key=value line is prefixed with ``#`` so it
1129    acts as a placeholder the user can uncomment.
1130    """
1131    help_text = _config_comment(action)
1132    if help_text:
1133        for chunk in textwrap.wrap(help_text, width=76,
1134                                   initial_indent="# ",
1135                                   subsequent_indent="# "):

1138        val_str = _format_config_value(action, value) if value is not None else ""
1139        lines.append(f"# {dest} = {val_str}\n")
1140    else:
1141        lines.append(f"{dest} = {_format_config_value(action, value)}\n")
1142    lines.append("\n")

1145def _print_settings_header(
1146    args: argparse.Namespace,
1147    args_cli: argparse.Namespace,
1148    parser: argparse.ArgumentParser,
1149    pure_defaults: dict,
1150    proj_ini_path: "Path | None",
1151    cfg_path: "Path | None",
1152    proj_ini_values: dict,
1153    cfg_values: dict,
1154) -> None:
1155    """Print the startup header: version, config files, non-default settings table.
1156
1157    Args:
1158        args: effective parsed namespace (defaults + ini + CLI).
1159        args_cli: parsed namespace with no ini defaults applied (CLI-only).
1160        parser: the CLI parser (for action metadata).
1161        pure_defaults: ``{dest: default}`` captured before any ``set_defaults``.
1162        proj_ini_path: path to the auto-loaded project ini, or ``None``.
1163        cfg_path: path to the ``--config`` ini, or ``None``.
1164        proj_ini_values: settings loaded from the project ini.
1165        cfg_values: settings loaded from ``--config``.
1166    """
1167    print(f"PyAutoRoute {__version__}")
1168    if proj_ini_path is not None:
1169        print(f"  Project ini:  {proj_ini_path}")
1170    if cfg_path is not None:
1171        print(f"  --config:     {cfg_path}")
1172
1173    actions = _configurable_actions(parser)
1174    rows: list[tuple[str, str, str]] = []
1175    for dest, action in actions.items():
1176        effective = getattr(args, dest, None)
1177        pure_def = pure_defaults.get(dest, action.default)
1178        if effective == pure_def:
1179            continue                        # still at default — skip
1180
1181        long_opt = next((s for s in action.option_strings if s.startswith("--")),
1182                        action.option_strings[0])
1183        val_str = _format_config_value(action, effective)
1184
1185        cli_val = getattr(args_cli, dest, pure_def)
1186        if cli_val != pure_def:
1187            source = "cli"
1188        elif dest in cfg_values and effective == cfg_values[dest]:
1189            source = cfg_path.name if cfg_path else "ini"
1190        elif dest in proj_ini_values:
1191            source = proj_ini_path.name if proj_ini_path else "ini"
1192        else:
1193            source = "ini"
1194
1195        rows.append((long_opt, val_str, source))
1196
1197    w_opt = max((len(r[0]) for r in rows), default=len("Option"))
1198    w_val = max((len(r[1]) for r in rows), default=len("Value"))
1199    w_opt = max(w_opt, len("Option"))
1200    w_val = max(w_val, len("Value"))
1201    if rows:
1202        hdr = f"  {'Option':<{w_opt}}  {'Value':<{w_val}}  Source"
1203        sep = f"  {'-' * w_opt}  {'-' * w_val}  ------"
1204        print()
1205        print(hdr)
1206        print(sep)
1207        for opt, val, src in rows:
1208            print(f"  {opt:<{w_opt}}  {val:<{w_val}}  {src}")
1209        print()
1210    sys.stdout.flush()

1213def write_config(parser: argparse.ArgumentParser, args, path: str | Path) -> None:
1214    """Write the effective settings to an INI file with per-option comments.
1215
1216    Every configurable option is included: options that have a value are
1217    written as ``key = value``; options that are unset (``None``) are written
1218    as ``# key =`` so they act as documented placeholders.  Within mutually
1219    exclusive groups only the active member is written un-commented; the
1220    alternatives appear as commented-out placeholders below it.
1221
1222    Args:
1223        parser: the CLI parser (for the option set and help strings).
1224        args: the parsed namespace whose effective values are written.
1225        path: destination settings-file path.
1226    """
1227    skip = _CONFIG_SKIP
1228    ordered = [(a.dest, a) for a in parser._actions
1229               if a.option_strings and a.dest not in skip]
1230    action_map = dict(ordered)
1231
1232    # Build a map from dest -> sibling dests for mutually exclusive groups.
1233    mutex_siblings: dict[str, list[str]] = {}
1234    for group in parser._mutually_exclusive_groups:
1235        dests = [a.dest for a in group._group_actions if a.dest not in skip]
1236        for dest in dests:
1237            mutex_siblings[dest] = [d for d in dests if d != dest]
1238
1239    lines = [
1240        "# PyAutoRoute settings — pass with --config FILE.\n",
1241        "# CLI options override these. Lists are comma-separated.\n",
1242        "\n",
1243        f"[{_CONFIG_SECTION}]\n",
1244        "\n",
1245    ]
1246    written: set[str] = set()
1247    for dest, action in ordered:
1248        if dest in written:
1249            continue
1250        written.add(dest)
1251        value = getattr(args, dest, None)
1252        _append_option(lines, dest, action, value, commented=(value is None))
1253
1254        # Immediately follow with any mutex siblings, always commented out.
1255        for sib in mutex_siblings.get(dest, []):
1256            if sib in written:
1257                continue
1258            written.add(sib)
1259            sib_val = getattr(args, sib, None)
1260            _append_option(lines, sib, action_map[sib], sib_val,
1261                           commented=True)
1262
1263    with open(path, "w") as f:
1264        f.writelines(lines)

1412def main(argv=None) -> int:
1413    """CLI entry point: parse arguments, validate them, and run the router.
1414
1415    Args:
1416        argv: argument list to parse; `None` uses ``sys.argv``.
1417
1418    Returns:
1419        The process exit code from `run` (0 clean, 2 on a self-check violation).
1420    """
1421    # Layer config sources so the final priority is:
1422    #   defaults  <  project ini  <  --config  <  CLI options
1423    # The project ini is <input_stem>.ini beside the board file and is loaded
1424    # automatically when present (no flag needed).  --config overrides it.
1425    pre = argparse.ArgumentParser(add_help=False)
1426    pre.add_argument("--config")
1427    pre.add_argument("input", nargs="?")
1428    known, _ = pre.parse_known_args(argv)
1429    parser = build_parser()
1430
1431    # Capture pure argparse defaults before any set_defaults calls.
1432    pure_defaults = {a.dest: a.default for a in parser._actions
1433                     if a.option_strings and a.dest not in _CONFIG_SKIP}
1434
1435    proj_ini_path: Path | None = None
1436    proj_ini_values: dict = {}
1437    if known.input:
1438        proj_ini = Path(known.input).with_suffix(".ini")
1439        d = load_project_config(proj_ini, parser)
1440        if d:
1441            proj_ini_path = proj_ini
1442            proj_ini_values = d
1443            parser.set_defaults(**d)
1444
1445    cfg_path: Path | None = None
1446    cfg_values: dict = {}
1447    if known.config:
1448        cfg_values = load_config(known.config, parser)
1449        cfg_path = Path(known.config)
1450        parser.set_defaults(**cfg_values)
1451
1452    args = parser.parse_args(argv)
1453
1454    # CLI-only namespace (no ini defaults) used for source detection in header.
1455    args_cli = build_parser().parse_args(argv)
1456
1457    if args.write_config is not None:
1458        cfg_path = (Path(args.write_config) if args.write_config
1459                    else Path(args.input).with_suffix(".ini"))
1460        write_config(parser, args, cfg_path)
1461        print(f"wrote settings to {cfg_path}")
1462        return 0
1463
1464    t_start, t_end = args.anneal_temps
1465    if not (t_start > t_end > 0):
1466        parser.error("--anneal-temps requires START > END > 0")
1467    if args.unrouted_weight < 0:
1468        parser.error("--unrouted-weight must be >= 0")
1469    if args.search_margin is not None and args.search_margin < 0:
1470        parser.error("--search-margin must be >= 0")
1471    if args.place_margin < 0:
1472        parser.error("--place-margin must be >= 0")
1473    if args.place_buffer is not None and args.place_buffer < 0:
1474        parser.error("--place-buffer must be >= 0")
1475    pt_start, pt_end = args.place_temps
1476    if not (pt_start > pt_end > 0):
1477        parser.error("--place-temps requires START > END > 0")
1478    if args.place_step <= 0:
1479        parser.error("--place-step must be > 0")
1480    if args.runs < 1:
1481        parser.error("--runs must be >= 1")
1482    if args.jobs < 0:
1483        parser.error("--jobs must be >= 0 (0 means use every CPU)")
1484    if args.place_runs < 1:
1485        parser.error("--place-runs must be >= 1")
1486    if (args.place_iters or args.place_time) and not (args.place or args.place_only):
1487        parser.error("--place-iters/--place-time require --place or --place-only")
1488    if args.place_only and (args.iters or args.time_budget):
1489        parser.error("--place-only does not route; drop --iters/--time")
1490    _print_settings_header(args, args_cli, parser, pure_defaults,
1491                           proj_ini_path, cfg_path,
1492                           proj_ini_values, cfg_values)
1493    return run(args, _print_version=False)

 40
 41    If ``log_path`` is given, every phase and a (throttled) trace of routing /

 48                 log_path: Path | None = None):
 49        self.stream = stream

 52        self._t0 = time.time()

 54        self.log_file = open(log_path, "w") if log_path else None

 57    def phase(self, name: str) -> None:
 58        """Announce a new phase on the display and in the log.
 59
 60        Args:
 61            name: the phase name, e.g. ``"routing 94 connections"``.
 62        """
 63        self.log(f"{name} ...")
 64        if self.quiet:

 66        self._write(f"[{self._elapsed():6.1f}s] {name} ...")
 67        if not self.tty:

 71        """Report greedy-routing progress.
 72
 73        Args:
 74            done: connections attempted so far.
 75            total: total connections to route.
 76            routed: count routed successfully.
 77            unrouted: count that failed.
 78        """
 79        msg = (f"{self.tag}routing {done}/{total}  routed={routed} failed={unrouted}")
 80        if done == total or done % 10 == 0:
 81            self.log(msg)
 82        if self.quiet:
 83            return
 84        line = f"[{self._elapsed():6.1f}s] {msg}"
 85        if self.tty:
 86            self._write(line)

 88            self.stream.write(line + "\n")
