pyautoroute.pcb

Board model loaded from a .kicad_pcb s-expression, plus a routed-board writer.

Parsing covers what the router needs: the copper layer stack, every pad with its absolute position/rotation/shape, the net-reference style (name-only as in KiCad 10, or a numbered net table as in KiCad 6-9), existing tracks/vias/zones to treat as obstacles, the free (dangling) vias to strip, and the Edge.Cuts outline shapes.

The writer clones the parsed tree, drops the free vias, appends freshly-built (segment ...) / (via ...) nodes, and serializes. Untouched subtrees keep their source spans so the diff against the input is limited to the routing edits.

 56def rotate(x: float, y: float, angle_deg: float) -> tuple[float, float]:
 57    """Rotate a point using KiCad's RotatePoint convention (matches pcbnew)."""
 58    a = math.radians(angle_deg)
 59    s, c = math.sin(a), math.cos(a)
 60    return (x * c + y * s, -x * s + y * c)

140def _net_name(net_node: SList | None, numbered: dict[int, str]) -> str:
141    """Resolve a ``(net ...)`` reference to a net name across both file styles."""
142    if net_node is None:
143        return ""
144    items = atoms_after_head(net_node)
145    if not items:
146        return ""
147    if len(items) == 1:
148        a = items[0]
149        if a.is_string:
150            return a.text                       # name-only: (net "GND")
151        return numbered.get(int(a.as_float()), "")   # numbered: (net 3)
152    # (net 3 "GND")
153    return items[-1].text

158def _copper_layers(tree: SList) -> list[str]:
159    layers_node = child(tree, "layers")
160    out: list[str] = []
161    if layers_node is None:
162        return ["F.Cu", "B.Cu"]
163    for entry in layers_node:
164        if not isinstance(entry, SList):
165            continue
166        # (0 "F.Cu" signal ...)
167        toks = [it for it in entry]
168        if len(toks) >= 3 and isinstance(toks[1], Atom):
169            name = toks[1].text
170            if name.endswith(".Cu"):
171                out.append(name)
172    return out or ["F.Cu", "B.Cu"]
173
174
175def _resolve_pad_layers(layer_strings: list[str], copper: list[str]) -> list[str]:
176    out: list[str] = []
177    for ls in layer_strings:
178        if ls == "*.Cu":
179            return list(copper)
180        if ls.endswith(".Cu") and ls in copper:
181            out.append(ls)
182    return out

185# --- pad parsing -------------------------------------------------------------
186
187def _parse_pad(pad_node: SList, fx: float, fy: float, fa: float,
188               copper: list[str], numbered: dict[int, str], fp_ref: str) -> Pad | None:
189    head = atoms_after_head(pad_node)
190    # (pad "1" smd roundrect ...) -> [number, type, shape]
191    if len(head) < 3:
192        return None
193    pad_type = head[1].raw
194    shape = head[2].raw
195
196    at = floats(child(pad_node, "at"))
197    px, py = (at + [0.0, 0.0])[:2]
198    pad_angle = at[2] if len(at) >= 3 else 0.0
199
200    size = floats(child(pad_node, "size"))
201    w, h = (size + [0.0, 0.0])[:2]
202
203    rx, ry = rotate(px, py, fa)
204    cx, cy = fx + rx, fy + ry
205
206    layers = _resolve_pad_layers(strings(child(pad_node, "layers")), copper)
207
208    rratio_node = child(pad_node, "roundrect_rratio")
209    rratio = rratio_node[1].as_float() if rratio_node else None
210
211    delta = floats(child(pad_node, "rect_delta"))
212    rect_delta = (delta[0], delta[1]) if len(delta) >= 2 else None
213
214    drill_node = child(pad_node, "drill")
215    drill = None
216    if drill_node is not None:
217        df = floats(drill_node)
218        if df:
219            drill = df[-1]   # (drill d) or (drill oval dx dy)
220
221    net = _net_name(child(pad_node, "net"), numbered)
222
223    return Pad(
224        net=net, pad_type=pad_type, shape=shape,
225        cx=cx, cy=cy, w=w, h=h, angle=pad_angle,
226        copper_layers=layers, roundrect_rratio=rratio,
227        rect_delta=rect_delta, drill=drill, fp_ref=fp_ref,
228    )

231def _footprint_ref(fp_node: SList) -> str:
232    for prop in children(fp_node, "property"):
233        vals = atoms_after_head(prop)
234        if vals and vals[0].text == "Reference" and len(vals) >= 2:
235            return vals[1].text
236    return ""

239# --- outline parsing ---------------------------------------------------------

241def _parse_outline(tree: SList) -> list[OutlineShape]:
242    shapes: list[OutlineShape] = []
243    for it in tree:
244        if not isinstance(it, SList):
245            continue
246        head = sexpr.head_symbol(it)
247        if head not in ("gr_poly", "gr_line", "gr_rect", "gr_arc", "gr_circle"):
248            continue
249        layer = strings(child(it, "layer"))
250        if not layer or layer[0] != "Edge.Cuts":
251            continue
252        if head == "gr_poly":
253            pts = _pts(child(it, "pts"))
254            shapes.append(OutlineShape("poly", {"pts": pts}))
255        elif head == "gr_line":
256            shapes.append(OutlineShape("line", {
257                "start": tuple(floats(child(it, "start"))),
258                "end": tuple(floats(child(it, "end"))),
259            }))
260        elif head == "gr_rect":
261            shapes.append(OutlineShape("rect", {
262                "start": tuple(floats(child(it, "start"))),
263                "end": tuple(floats(child(it, "end"))),
264            }))
265        elif head == "gr_arc":
266            shapes.append(OutlineShape("arc", {
267                "start": tuple(floats(child(it, "start"))),
268                "mid": tuple(floats(child(it, "mid"))),
269                "end": tuple(floats(child(it, "end"))),
270            }))
271        elif head == "gr_circle":
272            shapes.append(OutlineShape("circle", {
273                "center": tuple(floats(child(it, "center"))),
274                "end": tuple(floats(child(it, "end"))),
275            }))
276    return shapes

279def _pts(pts_node: SList | None) -> list[tuple[float, float]]:
280    if pts_node is None:
281        return []
282    out = []
283    for xy in pts_node:
284        if isinstance(xy, SList) and sexpr.head_symbol(xy) == "xy":
285            f = floats(xy)
286            if len(f) >= 2:
287                out.append((f[0], f[1]))
288    return out

293def _parse_free_vias(tree: SList, numbered: dict[int, str]) -> list[Via]:
294    out = []
295    for it in children(tree, "via"):
296        at = floats(child(it, "at"))
297        size = floats(child(it, "size"))
298        drill = floats(child(it, "drill"))
299        lays = strings(child(it, "layers"))
300        out.append(Via(
301            cx=at[0] if at else 0.0, cy=at[1] if len(at) > 1 else 0.0,
302            size=size[0] if size else 0.6,
303            drill=drill[0] if drill else 0.3,
304            layers=(lays[0], lays[-1]) if lays else ("F.Cu", "B.Cu"),
305            net=_net_name(child(it, "net"), numbered),
306            node=it,
307        ))
308    return out

311def _parse_segments(tree: SList, numbered: dict[int, str]) -> list[Segment]:
312    out = []
313    for it in children(tree, "segment"):
314        s = floats(child(it, "start"))
315        e = floats(child(it, "end"))
316        w = floats(child(it, "width"))
317        lay = strings(child(it, "layer"))
318        if len(s) >= 2 and len(e) >= 2:
319            out.append(Segment(
320                x1=s[0], y1=s[1], x2=e[0], y2=e[1],
321                width=w[0] if w else 0.2,
322                layer=lay[0] if lay else "F.Cu",
323                net=_net_name(child(it, "net"), numbered),
324            ))
325    return out

327
328def _parse_zones(tree: SList, numbered: dict[int, str]) -> list[dict]:
329    out = []
330    for it in children(tree, "zone"):
331        out.append({
332            "net": _net_name(child(it, "net_name"), numbered)
333                   or _net_name(child(it, "net"), numbered),
334            "layers": strings(child(it, "layers")) or strings(child(it, "layer")),
335            "polygon": _pts(child(child(it, "polygon"), "pts") if child(it, "polygon") else None),
336        })
337    return out

340def _numbered_net_table(tree: SList) -> dict[int, str]:
341    """Top-level (net N "name") declarations (KiCad 6-9). Empty for name-only."""
342    table: dict[int, str] = {}
343    for it in children(tree, "net"):
344        a = atoms_after_head(it)
345        if len(a) >= 2 and not a[0].is_string:
346            table[int(a[0].as_float())] = a[1].text
347    return table

352def load_board(pcb_path: str | Path) -> Board:
353    tree = sexpr.loads(Path(pcb_path).read_text())
354    copper = _copper_layers(tree)
355    numbered = _numbered_net_table(tree)
356    name_only = len(numbered) == 0
357
358    pads: list[Pad] = []
359    for fp in children(tree, "footprint"):
360        at = floats(child(fp, "at"))
361        fx, fy = (at + [0.0, 0.0])[:2]
362        fa = at[2] if len(at) >= 3 else 0.0
363        ref = _footprint_ref(fp)
364        for pad_node in children(fp, "pad"):
365            p = _parse_pad(pad_node, fx, fy, fa, copper, numbered, ref)
366            if p is not None:
367                pads.append(p)
368
369    return Board(
370        tree=tree,
371        copper_layers=copper,
372        pads=pads,
373        free_vias=_parse_free_vias(tree, numbered),
374        segments=_parse_segments(tree, numbered),
375        zones=_parse_zones(tree, numbered),
376        outline=_parse_outline(tree),
377        numbered_nets=numbered,
378        name_only_nets=name_only,
379    )

384def _net_ref_node(board: Board, net: str) -> SList:
385    """Build a ``(net ...)`` node in the board's own reference style."""
386    node = SList([sexpr.sym("net")])
387    if board.name_only_nets:
388        node.append(sexpr.string(net))
389    else:
390        code = next((n for n, name in board.numbered_nets.items() if name == net), 0)
391        node.append(sexpr.number(code))
392    return node

423def write_board(board: Board, out_path: str | Path,
424                new_nodes: list[SList] | None = None,
425                strip_free_vias: bool = True) -> None:
426    """Serialize a routed copy: drop free vias, append new segment/via nodes."""
427    strip_ids = {id(v.node) for v in board.free_vias} if strip_free_vias else set()
428    new_root = SList()
429    for ch in board.tree:
430        if isinstance(ch, SList) and id(ch) in strip_ids:
431            continue
432        new_root.append(ch)
433    for node in (new_nodes or []):
434        new_root.append(node)
435    Path(out_path).write_text(sexpr.dump_file(new_root))
